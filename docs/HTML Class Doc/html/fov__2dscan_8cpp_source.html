522 void FOV_2DScan::slotConfigChanged()
523 {
524  // Calculate the maximum displayed range and angular span
525 
526  int receiverQty = AWLSettings::GetGlobalSettings()->receiverSettings.size();
527  config.maxSensorsRange = 0.0;
528  config.maxAngularSpan = 0.0;
529  config.spareDepth = 0;
530 
531  for (int receiverID = 0; receiverID < receiverQty; receiverID++)
532  {
533  RelativePosition receiverPosition = SensorCoordinates::GetReceiverPosition(receiverID);
534  config.maxSensorsRange = std::max(config.maxSensorsRange, AWLSettings::GetGlobalSettings()->receiverSettings[receiverID].displayedRangeMax);
535 
536  config.spareDepth = std::min(config.spareDepth, -receiverPosition.position.bodyRelative.forward);
537 
538  int voxelQty = AWLSettings::GetGlobalSettings()->receiverSettings[receiverID].voxelsConfig.size();
539  for (int voxelIndex = 0; voxelIndex < voxelQty; voxelIndex++)
540  {
541  ReceiverSettings receiverSettings = AWLSettings::GetGlobalSettings()->receiverSettings[receiverID];
542  RelativePosition voxelPosition = SensorCoordinates::GetVoxelPosition(receiverID, voxelIndex);
543  VoxelConfig voxelConfig = receiverSettings.voxelsConfig[voxelIndex];
544 
545  float startAngle = RAD2DEG(receiverPosition.orientation.yaw) +
546  RAD2DEG(voxelPosition.orientation.yaw) + (voxelConfig.fovWidth / 2);
547  config.maxAngularSpan = std::max(config.maxAngularSpan, fabs(startAngle));
548  config.maxAngularSpan = std::max(config.maxAngularSpan, fabs(startAngle - voxelConfig.fovWidth));
549  }
550  }
551 
552  if (displayZoomMode == eDisplayZoomModeAuto)
553  config.spareDepth -= AWLSettings::GetGlobalSettings()->receiverSettings[0].displayedRangeMin;
554 
555  config.maxAngularSpan *= 2;
556  // Span is always a multiple of 10 degrees - just for display aethetics
557  config.maxAngularSpan = (1+ ((int)config.maxAngularSpan / 10)) * 10;
558 
559  setMinimumSize(minimumSizeHint());
560  calculateResize();
561 
562  update();
563 }

600 void FOV_2DScan::calculateResize()
601 {
602 
603  // For default mode where we are eDisplayZommModeFront
604  float totalDistance = config.maxSensorsRange+config.spareDepth;
605 //printf("MaxRange: %f %d\n", config.maxSensorsRange, config.spareDepth);
606  if (config.maxSensorsRange+config.spareDepth < 0.000000000000000001) return;
607 
608  if (displayZoomMode == eDisplayZoomMode360)
609  totalDistance = (config.maxSensorsRange * 2) + carLength;
610 
611  float maxWidth = width() + 1;
612  float maxHeight = height();
613 
614 
615  while (maxWidth > width())
616  {
617  Ratio = (maxHeight-topInPixels) / totalDistance;
618 
619  float angleInRad = DEG2RAD((config.maxAngularSpan/2)+180);
620  float xWidth = abs((totalDistance*Ratio)*sinf(angleInRad));
621 //printf("Ratio0: %f %f %f \n", totalDistance, Ratio, angleInRad);
622 //printf("Width0: %f %d %f \n", maxWidth, width(), xWidth);
623 
624  maxWidth = (xWidth*2)+rightInPixels;
625  if (maxWidth > width())
626  {
627  maxHeight = (maxHeight * (width() / maxWidth)) -1;
628  }
629 //printf("Width1: %f %d %f \n", maxWidth, width(), xWidth);
630  }
631 
632  if (displayZoomMode == eDisplayZoomModeFront || displayZoomMode == eDisplayZoomModeAuto)
633  {
634  zeroY = maxHeight - (config.spareDepth * Ratio);
635  zeroX = width()/2;
636  }
637  else
638  {
639  zeroY = maxHeight * 0.5; // .1 x height offcenter to the bottom
640  zeroX = width() / 2;
641  }
642 
643  int labelWidth = width() * 0.3;
644  int labelHeight = labelWidth;
645  logoLabel->resize(labelWidth, labelHeight);
646 
647  QPixmap *myPix = NULL;
648  if (!AWLSettings::GetGlobalSettings()->sLogoFileName.empty())
649  {
650  myPix = new QPixmap(AWLSettings::GetGlobalSettings()->sLogoFileName.c_str());
651  }
652 
653  if (myPix && !myPix->isNull())
654  {
655 
656  logoLabel->setPixmap((*myPix).scaled(labelWidth, labelHeight, Qt::KeepAspectRatio, Qt::SmoothTransformation));
657  }
658 
659 
660  logoLabel->move(logoLeftMargin, height() - labelHeight);
661 }

663 void FOV_2DScan::paintEvent(QPaintEvent * /*paintEvent*/)
664 {
665 #ifdef USE_FPS_FOV_2DSCAN
666  ++nFrames;
667 
668  // Record time
669  auto t1 = boost::chrono::high_resolution_clock::now();
670 #endif
671 
672  QPainter painter(this);
673 
674  // Draw ruler
675 
676  if (measureMode != eMeasureCartesian)
677  {
678  //Angular Ruler
679  painter.setPen(QPen(rgbRuler));
680  float angleIncrement = 5.0;
681  float distanceIncrement = 1.0;
682  if (config.maxSensorsRange >= 100.0) distanceIncrement = 10;
683  else if (config.maxSensorsRange > 10.0) distanceIncrement = 5;
684  else distanceIncrement = 1.0;
685 
686  for (float i = config.maxSensorsRange; i > 0; i-=distanceIncrement)
687  {
688  // All distances are relative to bumper
689  drawArc(&painter, -config.maxAngularSpan/2, config.maxAngularSpan, i);
690  }
691 
692  painter.setPen(QPen(rgbRuler));
693  for (float i = -config.maxAngularSpan / 2; i <= config.maxAngularSpan / 2; i += angleIncrement)
694  {
695  drawLine(&painter, i, 0, config.maxSensorsRange);
696  }
697 
698  // Draw the distance indicators
