64 AWLSimpleDemo::AWLSimpleDemo()
65 {
66 
67  // Read the settigs from the configuration file
68  AWLSimpleSettings *globalSettings = AWLSimpleSettings::InitSettings();
69  globalSettings->ReadSettings();
70 
71  // Build a reference coodinate system from the settings
72  SensorCoordinates *globalCoordinates = SensorCoordinates::InitCoordinates();
73  globalCoordinates->BuildCoordinatesFromSettings(globalSettings->GetPropTree());
74 
75  // Create the receiver communication objects
76  int receiverQty = globalSettings->receiverSettings.size();
77  receiverCaptures.resize(receiverQty);
78  for (int receiverID = 0; receiverID < receiverQty; receiverID++)
79  {
80  // Create the LIDAR acquisition thread object, depending on the type identified in the config file
81 
82 
83 
84 #ifdef USE_LIBUSB
85  if (globalSettings->receiverSettings[receiverID].sReceiverType == std::string("LibUSB"))
86  {
87  // LibUSB Capture is used if defined in the ini file
88  receiverCaptures[receiverID] = ReceiverCapture::Ptr(new ReceiverLibUSBCapture(receiverID, globalSettings->GetPropTree()));
89  }
90  else
91 #endif
92  {
93  // If the type is undefined, just use the dumb simulator, not using external device
94  receiverCaptures[receiverID] = ReceiverCapture::Ptr(new ReceiverSimulatorCapture(receiverID, globalSettings->GetPropTree()));
95  }
96 
97  receiverCaptureSubscriberIDs.push_back(receiverCaptures[receiverID]->Subscribe());
98  }
99 
100 
101 }

111 int AWLSimpleDemo::exec()
112 
113 {
114  // Start the threads for background receiver capture objects
115  for (size_t receiverID = 0; receiverID < receiverCaptures.size(); receiverID++)
116  {
117  receiverCaptures[receiverID]->Go();
118  }
119 
120 
121  // And loop to display all the detections like crazy
122  while (!_kbhit())
123  {
124  DoOneLoopIteration();
125  }
126 
127  return(0);

186 bool AWLSimpleDemo::OutputDetections(const Detection::Vector &detectionData)
187 {
188  BOOST_FOREACH(const Detection::Ptr &detection, detectionData)
189  {
190  // Prepend the current time to output
191  boost::posix_time::ptime myTime(boost::posix_time::microsec_clock::local_time());
192 
193  std::string timeStr(boost::posix_time::to_simple_string(myTime));
194  timeStr += ";";
195 
196  // For mat the detection information
197  char detectionString[255];
198  sprintf_s(detectionString, 255, "Receiver %02d;column, row %02d %02d ;Distance from sensor %02.2f;Distance from car %02.2f", detection->receiverID, detection->cellID.column, detection->cellID.row, detection->relativeToSensorCart.cartesian.x, detection->relativeToVehicleCart.cartesian.x);
199 
200  timeStr += detectionString;
201  timeStr += "\n";
202 
203  // Output this to the display
204  printf(timeStr.c_str());
205  }
206 
207  return(true);
