72 void DebugFilePrintf(const char *format, ...)
73 {
74 
75  if (SensorSettings::GetGlobalSettings()->bWriteDebugFile)
76  {
77 
78  // Open the debug file
79  std::ofstream debugFile;
80  bool bOpen = OpenDebugFile(debugFile, SensorSettings::GetGlobalSettings()->sDebugFileName, true);
81  if( !bOpen) return;
82 
83 
84  // Format the string and output
85 
86  boost::posix_time::ptime myTime(boost::posix_time::microsec_clock::local_time());
87 
88  std::string timeStr(boost::posix_time::to_simple_string(myTime));
89  timeStr += " ";
90 
91  va_list argList;
92  va_start(argList, format);
93  size_t len = std::vsnprintf(NULL, 0, format, argList);
94  va_end(argList);
95 
96  std::vector<char> vec(len + 1);
97  va_start(argList, format);
98  std::vsnprintf(&vec[0], len + 1, format, argList);
99  va_end(argList);
100 
101 
102  timeStr += std::string(vec.data());
103  timeStr += "\n";
104 
105  debugFile << timeStr;
106  // Close the debug file
107  CloseDebugFile(debugFile);
108 
109  }
110 }

143 bool OpenDebugFile(std::ofstream &debugFile, const std::string filename, bool bAppend)
144 {
145  if (SensorSettings::GetGlobalSettings()->bWriteDebugFile)
146  {
147  if (debugFile.is_open())
148  {
149  CloseDebugFile(debugFile);
150  }
151 
152  std::string sFileName = SensorSettings::GetGlobalSettings()->sDebugAndLogFilePath + filename;
153 
154  if (bAppend)
155  {
156  debugFile.open(sFileName, std::ofstream::out | std::ofstream::app);
157  }
158  else
159  {
160  debugFile.open(sFileName);
161  }
162 
163  if ( debugFile.fail() )
164  {
165  std::string sErr = " Failed to open ";
166  sErr += filename;
167  fprintf(stderr, sErr.c_str());
168 
169  return (false);
170  }
171  }
172  return(true);
173 }
