73 uint32_t numberOfSetBits(uint32_t i)
74 {
75  // Java: use >>> instead of >>
76  // C or C++: use uint32_t
77  i = i - ((i >> 1) & 0x55555555);
78  i = (i & 0x33333333) + ((i >> 2) & 0x33333333);
79  return (((i + (i >> 4)) & 0x0F0F0F0F) * 0x01010101) >> 24;
80 }

130 void AWLPlotScan::LabelAScan(QPainter* p)
131 {
132  float fAscanHeight = float(height()) / (m_nbrCh + 1);
133 
134  float maxRange, scale;
135  int step;
136  AWLSettings *globalSettings = AWLSettings::GetGlobalSettings();
137  maxRange = globalSettings->receiverSettings[0].voxelsConfig[0].maxAscanRange / 5;
138  if ( maxRange < 10.0 ) {
139  step = ((int)((maxRange+2.5)/5)) * 5;
140  } else {
141  step = ((int)((maxRange+5)/10)) * 10;
142  }
143 
144  scale = width() * ( (float) step / maxRange );
145 
146  p->setPen(QPen(rgbRulerMed));
147  p->setBrush(QBrush(rgbRulerMed));
148 
149  p->drawText(scale*1/5, 10, QString::number(step*1) + "m");
150  p->drawText(scale*2/5, 10, QString::number(step*2) + "m");
151  p->drawText(scale*3/5, 10, QString::number(step*3) + "m");
152  p->drawText(scale*4/5, 10, QString::number(step*4) + "m");
153 
154  //p->drawLine(0, SCAN_GRID_ORIGIN , width(), 15);
155  p->drawLine(scale*1/5, SCAN_GRID_ORIGIN , scale*1/5, 17 * fAscanHeight);

157  p->drawLine(scale*3/5, SCAN_GRID_ORIGIN , scale*3/5, 17 * fAscanHeight);

161 void AWLPlotScan::PlotAScan(QPainter* p, AScan::Ptr pAscan, int top, int left, int width, int height, float maxRange)
162 {
163  int16_t *b16;
164  int32_t *b32;
165  float xScaleFactor = 0.0;
166  float yScaleFactor = 0.0;
167  int32_t x1, y1;
168  int i;
169 
170  p->setPen(QPen(rgbRulerLight));
171  p->setBrush(QBrush(rgbRulerMed));
172 
173  if (pAscan->samples) {
174  x1 = left;
175  y1 = top;
176  //printf ("%d %d\n", sampleCount, width);
177  if (pAscan->sampleCount > static_cast<size_t>(width)) {
178  if (width > m_numPts)
179  {
180  m_numPts = width;
181  m_pPts = (QPoint*) malloc(m_numPts * sizeof(QPoint));
182  }
183 
184  if (width)
185  xScaleFactor = (float)pAscan->sampleCount / (float)width;
186 
187  yScaleFactor = height / maxRange;
188 
189  b16 = (int16_t *)(pAscan->samples);
190  b32 = (int32_t *)(pAscan->samples);
191  for (int x = 0; x < width; x++) {
192  x1 = left + x;
193  i = x * xScaleFactor;
194  switch (pAscan->sampleSize) {
195  default:
196  return;
197  case 2:
198  y1 = top - b16[i + pAscan->sampleOffset] * yScaleFactor;
199  break;
200  case 4:
201  y1 = top - b32[i + pAscan->sampleOffset] * yScaleFactor;
202  break;
203  }
204  m_pPts[x].setX(x1);
205  m_pPts[x].setY(y1);
206  }
207 
208  p->drawPolyline(m_pPts, width);
209  }
210  else {
211  if (pAscan->sampleCount > static_cast<size_t>(m_numPts))
212  {
213  m_numPts = pAscan->sampleCount;
214  m_pPts = (QPoint*)malloc(m_numPts * sizeof(QPoint));
215  }
216 
217  if (pAscan->sampleCount)
218  xScaleFactor = (float)width / (float)pAscan->sampleCount;
219 
220  yScaleFactor = height / maxRange;
221 
222  b16 = (int16_t *)(pAscan->samples);
223  b32 = (int32_t *)(pAscan->samples);
224 
225  for (size_t x = 0; x < pAscan->sampleCount; x++) {
226  i = x * xScaleFactor;
227  x1 = left + i;
228  switch (pAscan->sampleSize) {
229  default:
230  return;
231  case 2:
232  y1 = top - (b16[x + pAscan->sampleOffset] * yScaleFactor);
233  break;
234  case 4:
235  y1 = top - (b32[x + pAscan->sampleOffset] * yScaleFactor);
236  break;
237  }
238  m_pPts[x].setX(x1);
239  m_pPts[x].setY(y1);
240  }
241 
242  p->drawPolyline(m_pPts, pAscan->sampleCount);
243  }
244  }

256 void AWLPlotScan::plotAScans(QPainter* p)
257 {
258  int i = 0;
259  int chIdx = 0;
260 
261  if (!showAScan) return;
262 
263 #ifdef USE_FPS_AWLPLOTSCAN
264  ++nFrames;
265 #endif //USE_FPS_AWLPLOTSCAN
266 
267  float minFinal = FLT_MAX;
268  float maxFinal = -FLT_MAX;
269  float maxRange = 0.0F;
270  float fAscanHeight = float(height()) / (m_nbrCh + 1);
271 
272  // Text formatting info
273  int lineSpacing = p->fontMetrics().lineSpacing();
274  int leading = p->fontMetrics().leading();
275 
276  if (m_maxRange)
277  {
278  maxRange = m_maxRange;
279  }
280  else
281  {
282  i = 0;
283  BOOST_FOREACH(const AScan::Ptr & aScan, aScanData)
284  {
285  if (aScan->receiverID == m_selectedReceiver && m_chMask & (1 << aScan->cellID.column))
286  {
287  float minV, maxV, meanV;
288 
289  aScan->FindMinMaxMean(&minV, &maxV, &meanV);
290 
291  if (minV < minFinal)
292  minFinal = minV;
293 
294  if (maxV > maxFinal)
295  maxFinal = maxV;
296  }
297 
298  ++i;
299  }
300 
301  // maxRange will contain the peak absolute value for all ascans
302  maxRange = abs(maxFinal);
303 
304  if (abs(minFinal) > maxRange)
305  maxRange = abs(minFinal);
306  }
307 
308  LabelAScan(p);
309 
310  i = 0;
311  BOOST_FOREACH(const AScan::Ptr & aScan, aScanData)
312  {
313  if (aScan->receiverID == m_selectedReceiver && m_chMask & (1 << aScan->cellID.column))
314  {
315  PlotAScan(p, aScan, fAscanHeight * (chIdx + 1), 0, width(), fAscanHeight, maxRange);
316 
317  p->setBrush(QBrush(rgbRulerMed));
318  p->setPen(QPen(rgbRulerText));
319  p->drawText(SCAN_POSX, fAscanHeight * (chIdx + 1) - (leading/2) - 1, "Rcv " + QString::number(aScan->receiverID + 1) + " Col " + QString::number(aScan->cellID.column));
320  p->drawText(SCAN_POSX, fAscanHeight * (chIdx + 1) + lineSpacing - (leading/2) -1, "Channel " + QString::number(aScan->channelNo));
321  p->drawLine(SCAN_POSX, fAscanHeight * (chIdx + 1), width(), fAscanHeight * (chIdx + 1));
322 
323  ++chIdx;
324  }
325 
326  ++i;
327  }
328 
329 #ifdef USE_FPS_AWLPLOTSCAN
330  boost::chrono::duration<double> elapsed = t1 - m_timeFPS;
331 
332  if (elapsed.count() > 1.0)
333  {
334  FPS = nFrames;
335  nFrames = 0;
336  m_timeFPS = t1;
337  }
338 
339  p->setPen(QPen(rgbRulerLight));
340  p->setBrush(QBrush(rgbRulerMed));
341  p->drawText(10, 10, QString::number(FPS) + " FPS");
342 #endif //USE_FPS_AWLPLOTSCAN
343 
344 }
