107 bool ReceiverPolledCapture::LidarQuery(size_t& cycleCount, size_t& messageCount)
108 {
109  int transferred = 0;
110  int received = 0;
111  uint32_t dwCount;
112  uint32_t dwReadPending;
113 
114  boost::mutex::scoped_lock rawLock(m_Mutex);
115 
116  if (!handle)
117  return false;
118 
119  ReceiverCANMessage msg;
120  msg.id = RECEIVERCANMSG_ID_LIDARQUERY;
121 
122  ReceiverCANMessage resp;
123 
124  transferred = WriteBytes((uint8_t*)&msg, sizeof(msg));
125  if (transferred != sizeof(ReceiverCANMessage))
126  return false;
127 
128  received = ReadBytes((uint8_t*)&resp, sizeof(resp));
129  if (received != sizeof(ReceiverCANMessage))
130  return false;
131 
132  dwCount = *((uint32_t*)&resp.data[0]);
133  dwReadPending = *((uint32_t*)&resp.data[4]);
134 
135  cycleCount = (size_t)dwCount;
136  messageCount = (size_t)dwReadPending;
137 
138  return true;
139 }

167 bool ReceiverPolledCapture::DoOneLoop()
168 {
169  size_t cycleCount = 0;
170  size_t messageCount = 0;
171 
172  bool bRet = LidarQuery(cycleCount, messageCount);
173 
174  if (!bRet)
175  return false;
176 
177  if (cycleCount)
178  {
179  bool bReadSuccess = true;
180 
181  size_t payloadSize = cycleCount * sizeof(tDataFifo);
182  if (!xmitsFooterData)
183  {
184  payloadSize -= WAVEFORM_FOOTER_SIZE * sizeof(short);
185  }
186  bReadSuccess = ReadDataFromUSB((char*)dataFifo, (int) payloadSize, cycleCount);
187 
188 
189 
190  if (SensorSettings::GetGlobalSettings()->bWriteLogFile)
191  {
192  boost::mutex::scoped_lock rawLock(m_Mutex);
193 
194  if (SensorSettings::GetGlobalSettings()->bWriteLogFile) // Check again for file handle now that we have the mutex
195  {
196  for (size_t cycle = 0; cycle < cycleCount; cycle++)
197  {
198  LogWaveform((int) cycle);
199  }
200  }
201 
202  }
203 
204  ProcessRaw((uint8_t*)dataFifo->AcqFifo);
205  }
206 
207  if (messageCount)
208  {
209  bRet = PollMessages(messageCount);
210  if (!bRet)
211  return false;
212  }
213 
214  if (!cycleCount && !messageCount)
215  {
216  boost::this_thread::sleep(boost::posix_time::milliseconds(5));
217  }
218 
219 
220  return bRet;
221 }

223 void ReceiverPolledCapture::DoOneThreadIteration()
224 {
225  if (swap_handle) {
226  handle = swap_handle;
227  swap_handle = NULL;
228  }
229  if (handle)
230  {
231  bool bRet = DoOneLoop();
232 
233  if (!bRet)
234  {
235  CloseCANPort();
236  }
237  }
238  else
239  {
240  if (boost::posix_time::microsec_clock::local_time() > reconnectTime)
241  {
242  if (OpenCANPort())
243  {
244  WriteCurrentDateTime();
245  SetMessageFilters(receiverStatus.demandedFrameRate, receiverStatus.voxelMask, receiverStatus.messageMask);
246  // Update all the info (eventually) from the status of the machine
247  QueryAlgorithm();
248  QueryTracker();
249  }
250  }
251  }
252 }

262 bool ReceiverPolledCapture::PollMessages(size_t messageCount)
263 {
264  ReceiverCANMessage canResp[MAX_POLL_CAN_MESSAGES];
265  int transferred = 0;
266  int received = 0;
267 
268  boost::mutex::scoped_lock rawLock(m_Mutex);
269 
270  if (!handle)
271  return false;
272 
273  if (messageCount > MAX_POLL_CAN_MESSAGES)
274  messageCount = MAX_POLL_CAN_MESSAGES;
275 
276  ReceiverCANMessage msg;
277  msg.id = RECEIVERCANMSG_ID_POLLMESSAGES;
278  msg.data[0] = (unsigned char) messageCount;
279 
280  transferred = WriteBytes((uint8_t*)&msg, sizeof(msg));
281  if (transferred != sizeof(msg))
282  return false;
283 
284  received = ReadBytes((uint8_t*) &canResp[0], (int) (messageCount * sizeof(ReceiverCANMessage)));
285  if (received != (int) (messageCount * sizeof(ReceiverCANMessage)))
286  return false;
287 
288  for (uint32_t i=0; i<messageCount; i++)
289  {
290  if (canResp[i].id)
291  ParseMessage(canResp[i]);
292  }
293 
294  return true;
295 }

340 void ReceiverPolledCapture::LogWaveform(size_t cycle)
341 {
342  logFileMutex.lock();
343 
344  if (!logFilePtr)
345  {
346  logFileMutex.unlock();
347  return;
348  }
349 
350  for (int ch = 0; ch < CHANNEL_QTY; ch++)
351  {
352  CellID cellID = GetCellIDFromChannel(ch);
353  std::string theWaveString(", ,Wave, ,");
354  theWaveString += std::to_string(receiverID);
355  theWaveString += std::string(",Channel,");
356  theWaveString += std::to_string(cellID.column) + "," + std::to_string(cellID.row) + ",";
357  theWaveString += std::to_string(ch) + ", ,";
358 
359  short* pData = &dataFifo[cycle].AcqFifo[ch * WAVEFORM_POINT_QTY];
360  for (size_t i = 0; i < WAVEFORM_POINT_QTY; i++)
361  {
362  if (i == (WAVEFORM_POINT_QTY - 1))
363  theWaveString += std::to_string(*pData++);
364  else
365  theWaveString += std::to_string(*pData++) + ",";
366 
367  }
368  LogFilePrintf(*logFilePtr, theWaveString.c_str());
369  }
370 
371  short* pFooter = &dataFifo[cycle].footer[0];
372  std::string theFooterString(", ,Footer,,");
373  theFooterString += std::to_string(receiverID) + ",,,,";
374  for (size_t i = 0; i < WAVEFORM_FOOTER_SIZE; i++)
375  {
376  if (i == WAVEFORM_FOOTER_SIZE - 1) {
377  theFooterString += std::to_string(*pFooter++);
378  }
379  else
380  {
381  theFooterString += std::to_string(*pFooter++) + ",";
382  }
383  }
384  LogFilePrintf(*logFilePtr, theFooterString.c_str());
385 
386  logFileMutex.unlock();
387 }

462 void ReceiverPolledCapture::ProcessRaw(uint8_t* rawData)
463 {
464  size_t sampleOffset = 0;
465  size_t sampleDrop = 0;
466  size_t sampleSize = 1;
467  bool sampleSigned = false;
468 
469  uint16_t* rawData16;
470 
471  rawData16 = (uint16_t*)rawData;
472 
473  sampleOffset = 0;
474  sampleSize = 2;
475  sampleSigned = true;
476  sampleCount = 100;
477  sampleDrop = 1;
478 
479 
480  for (int channel = 0; channel < 16; channel++)
481  {
482  if (!rawBuffers[channel])
483  rawBuffers[channel] = new uint8_t[maxRawBufferSize];
484 
485  rawBufferCount++;
486 
487 
488  CellID cellID = GetCellIDFromChannel(channel);
489  memcpy(rawBuffers[channel], rawData + (channel * (100 * 2)), 100 * 2);
490  }
491 
492  boost::mutex::scoped_lock rawLock(GetMutex());
493 
494  for (int column = 0; column < 16; column++)
495  {
496  CellID cellID(column, 0);
497  int channelID = GetChannelIDFromCell(cellID);
498  AScan::Ptr aScan = currentFrame->MakeUniqueAScan(currentFrame->aScans, receiverID, cellID, channelID);
499  aScan->samples = rawBuffers[channelID];
500  aScan->sampleSize = sampleSize;
501  aScan->sampleOffset = sampleOffset;
502  aScan->sampleCount = sampleCount - sampleDrop;
503  aScan->sampleSigned = sampleSigned;
504  }
505 }
