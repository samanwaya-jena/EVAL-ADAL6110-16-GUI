204 
205 {
206  uint8_t buffer[receiveBufferSize];
207  int bytesRead;
208  bool sent;
209 
210  if (swap_handle) {
211  handle = swap_handle;
212  swap_handle = NULL;
213  }
214  if (handle)
215  {
216  sent = WriteMessage(pollMsg);
217 
218  ReceiverCANMessage* msgPtr = (ReceiverCANMessage*)buffer;
219 
220 #ifdef _WINDOWS_
221  // Under Windows, do not wait for other input.
222  bytesRead = ReadBytes((uint8_t*)buffer, sizeof(buffer),0);
223 #else
224  // Use time-out defined in XML file
225  bytesRead = ReadBytes((uint8_t*)buffer, sizeof(buffer));
226 #endif
227 
228 
229  if (bytesRead < sizeof(ReceiverCANMessage))
230  {
231  CloseCANPort();
232  }
233  else

235  if (msgPtr->id < 0xB0)
236  {

238  }
239  else
240  {
241  ProcessRaw(buffer);
242  }
243  }
244  }
245 
246  else { // Handle = NULL
247  if (boost::posix_time::microsec_clock::local_time() > reconnectTime) {
248  if (OpenCANPort()) {
249  QueryProductID();
250  QueryUniqueID();
251  WriteCurrentDateTime();
252  SetMessageFilters(receiverStatus.demandedFrameRate, receiverStatus.voxelMask, receiverStatus.messageMask);
253  }
254  }

259 bool ReceiverLibUSB2Capture::OpenCANPort()
260 {
261  int ret = 0;
262 
263  reconnectTime = boost::posix_time::microsec_clock::local_time() + boost::posix_time::milliseconds(reopenPortDelaylMillisec);
264 
265  ret = libusb_init(&context);
266 
267  if (ret)
268  return false;
269 
270  // discover devices
271  libusb_device** list;
272  libusb_device_descriptor descriptor;
273  ssize_t cnt = libusb_get_device_list(NULL, &list);
274  ssize_t i = 0;
275  int err = 0;
276  if (cnt < 0)
277  perror("libusb_get_device_list");
278 
279  int matches = 0;
280 
281  for (i = 0; i < cnt; i++) {
282  libusb_device* device = list[i];
283  err = libusb_get_device_descriptor(device, &descriptor);
284  if (!err && descriptor.idVendor == usbVendorId && descriptor.idProduct == usbProductId) {
285  matches++;
286  }
287  }
288 
289  //if (matches != (receiverID + 1)) return false;
290  if (matches < (receiverID + 1)) return false;
291 
292  for (i = 0; i < cnt; i++) {
293  libusb_device* device = list[i];
294  err = libusb_get_device_descriptor(device, &descriptor);
295  if (!err && descriptor.idVendor == usbVendorId && descriptor.idProduct == usbProductId) {
296  libusb_device_handle* h;
297  err = libusb_open(device, &h);
298  if (err) {
299  perror("libusb_open_no_access");
300  }
301  else {
302  handle = h;
303  //Claim Interface 0 from the device
304  ret = libusb_claim_interface((libusb_device_handle*)handle, 0);
305 
306  if (!ret) {
307  fprintf(stderr, "%ld %ld usb_claim_interface %ld %p succeeded\n", receiverID, matches, i, handle);
308  }
309  else {
310  fprintf(stderr, "%ld %ld usb_claim_interface %ld %p error %d\n", receiverID, matches, i, handle, ret);
311  libusb_close((libusb_device_handle*)handle);
312  handle = NULL;
313  }
314  }
315  }
316  if (handle) break;
317  }
318 
319  //printf("out %p\n", handle);
320  if (handle)
321  {
322  // force stop on acquisition. Unit may have been running for a while and
323  // has data in queue we may need to get rid of.
324  SetFPGARegister(acquisitionRegister, 0);
325  // Wait for the acquisition to stop
326  boost::this_thread::sleep(boost::posix_time::milliseconds(100));
327 
328  FlushMessages();
329 
330  // Restart the acquisition
331  SetFPGARegister(acquisitionRegister, 1);
332  return(true);
333  }
334  else { // if (!handle)
335  ClearAllRegisters();
336  return false;
337  }
338 }

361 bool ReceiverLibUSB2Capture::FlushMessages()
362 {
363  int bytesRead;
364 
365  uint8_t buffer[receiveBufferSize];
366 
367  if (!handle) return(false);
368 
369  // Clear any outstanding data left in the USB buffers
370  do
371  {
372  bytesRead = ReadBytes((uint8_t*)buffer, sizeof(buffer));
373  } while (bytesRead > 0);
374 
375  // Wait for a complete reset of the unit, in case this is where we start the unit
376  boost::this_thread::sleep(boost::posix_time::milliseconds(100));
377 
378  // Clear messages outstanding in the sensor's internal queues, but do not process them.
379  // End only when there is no more datam, or we have hit a QUEU_EMPTY message.
380  ReceiverCANMessage inMsg;
381  bool bEmpty = false;
382 
383  do
384  {
385  WriteMessage(pollMsg);
386  bytesRead = ReadBytes((uint8_t*)buffer, sizeof(buffer));
387  if (bytesRead <= sizeof(ReceiverCANMessage)) bEmpty = true;
388  else {
389  inMsg.id = buffer[0];
390  inMsg.len = buffer[9];
391  if (inMsg.id == RECEIVERCANMSG_ID_COMMANDMESSAGE && buffer[10] == RECEIVERCANMSG_ID_CMD_RESPONSE_QUEUE_EMPTY)
392  {
393  bEmpty = true;
394  }
395  }
396 
397  } while (!bEmpty);
398 
399  // Messages have been flushed. Reset the sensor status for comm errors.
400  receiverStatus.receiverError.byteData = 0;
401 
402  return true;
403

406 bool ReceiverLibUSB2Capture::SetMessageFilters(ReceiverFrameRate frameRate, VoxelMask voxelMask, MessageMask messageMask)
407 
408 {
409  ReceiverCANMessage message;
410  message.id = RECEIVERCANMSG_ID_COMMANDMESSAGE; // Message id: RECEIVERCANMSG_ID_COMMANDMESSAGE- Command message
411  message.len = RECEIVERCANMSG_LEN; // Frame size (0.8)
412 
413  bool bMessageOk = true;
414 
415  // Stop the acquisition while we set the operating parameters
416  bMessageOk = SetFPGARegister(acquisitionRegister, 0);
417  // Wait for the acquisition to stop
418  boost::this_thread::sleep(boost::posix_time::milliseconds(100));
419 
420  if (bMessageOk)
421  {
422  // Set Frame Rate
423  bMessageOk = SetFPGARegister(frameRateRegister, frameRate);
424  }
425 
426  if (bMessageOk)
427  {
428  message.data[0] = RECEIVERCANMSG_ID_CMD_TRANSMIT_COOKED; // Transmit_cooked enable flags
429 
430  *(int16_t*)&message.data[1] = voxelMask.wordData;
431  message.data[3] = 1; // Rate Decimation.
432  message.data[4] = 0; // Reserved
433  message.data[5] = 0; // Reserved
434  message.data[6] = 0; // Reserved
435  message.data[7] = 0; // Reserved
436 
437  bMessageOk = WriteMessage(message);
438  }
439 
440  if (bMessageOk)
441  {
442  message.data[0] = RECEIVERCANMSG_ID_CMD_TRANSMIT_RAW; // Transmit_raw enable flags
443 
444  if (messageMask.bitFieldData.raw)
445  {
446  *(int16_t*)&message.data[1] = voxelMask.wordData;
447  }
448  else
449  {
450  *(int16_t*)&message.data[1] = 0;
451  }
452  message.data[3] = 1; // Rate Decimation.
453  message.data[4] = 0; // Reserved
454  message.data[5] = 0; // Reserved
455  message.data[6] = 0; // Reserved
456  message.data[7] = 0; // Reserved
457  bMessageOk = WriteMessage(message);
458  }
459 
460 
461  // Restart the acquisition
462  if (bMessageOk)
463  {
464  bMessageOk = SetFPGARegister(acquisitionRegister, 1);
465  }
466 
467 
468  // The message has no confirmation built in
469  return(bMessageOk);
470 }

474 bool ReceiverLibUSB2Capture::WriteMessage(const ReceiverCANMessage& inMsg)
475 {
476 #if 1
477  WriteBytes((uint8_t *)&inMsg, sizeof(ReceiverCANMessage));
478 #else

484 
485  buf32 = (uint32_t*)buffer;

505  buffer[offset] = 0; //pad1
506  offset++;
507  buffer[offset] = 0; //pad2
508  offset++;
509  //printf("len : %d\n", offset);
510 
511  ret = WriteBytes((uint8_t*)buffer, offset);
512  // ret = sendto(fd, (char*)buffer, offset + inMsg.len, 0,
513  // (struct sockaddr *)remoteAddress, sizeof(sockaddr_in));
514 #endif
515  return(true);

572 void ReceiverLibUSB2Capture::ProcessRaw(uint8_t* rawData)
573 {
574  uint16_t* rawData16 = (uint16_t*)rawData;
575  int voxelIndex = rawData16[1] & 0xFF;
576  CellID cellID(voxelIndex, 0);
577  int msg_id = rawData[0];
578  bool transmit = false;
579 
580  if (msg_id != 0xb0) return;
581  if (voxelIndex >= maxRawBufferCount) return;
582 
583  if (!rawBuffers[voxelIndex])
584  {
585  rawBuffers[voxelIndex] = new uint8_t[maxRawBufferSize];
586  }
587  rawBufferCount++;
588 
589  memcpy(rawBuffers[voxelIndex], rawData, receiveBufferSize);
590 
591  sampleCount = (receiveBufferSize / 2) - sampleOffset;
592  transmit = true;
593 
594  if (voxelIndex > max_voxel) max_voxel = voxelIndex;
595  if (voxelIndex == max_voxel) transmit = true;
596 
597 
598  if (transmit)
599  {
600 
601  boost::mutex::scoped_lock rawLock(GetMutex());
602 
603  AScan::Ptr aScan = currentFrame->MakeUniqueAScan(currentFrame->aScans, receiverID, cellID, GetChannelIDFromCell(cellID));
604  aScan->samples = rawBuffers[voxelIndex];
605  aScan->sampleSize = sampleSize;
606  aScan->sampleOffset = sampleOffset;
607  aScan->sampleCount = sampleCount - sampleDrop;
608  aScan->sampleSigned = sampleSigned;
609  //printf("transmit ascan %d %d\n", aScan->channelID, aScan->sampleCount);
610 
611  rawLock.unlock();
612  }
613 
614  // Log the data, if logging is enabled.
615  if (SensorSettings::GetGlobalSettings()->bWriteLogFile)
616  {
617  boost::mutex::scoped_lock rawLock(m_Mutex);
618 
619  if (SensorSettings::GetGlobalSettings()->bWriteLogFile) // Check again for file handle now that we have the mutex
620  {
621  LogWaveform(rawData);
622  }
623  rawLock.unlock();
624  }
625 }
