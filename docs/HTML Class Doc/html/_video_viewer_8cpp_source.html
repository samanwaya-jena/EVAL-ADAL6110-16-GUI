214 void VideoViewer::slotImageChanged()
215 
216 {
217  // Update the video frame
218  cv::Mat tmpFrame;
219 
220  if (videoCapture != NULL && videoCapture->HasNews(currentVideoSubscriberID))
221  {
222 
223  VideoCapture::FramePtr cameraFrame(new (cv::Mat));
224 
225  // Copy the contents of the cv::Mat
226  videoCapture->CopyCurrentFrame(cameraFrame, currentVideoSubscriberID);
227  // Convert the image to the RGB888 format
228  int frameType = cameraFrame->type();
229  switch (frameType) {
230  case CV_8UC1:
231  cvtColor(*cameraFrame, tmpFrame, CV_GRAY2RGB);
232  break;
233  case CV_8UC3:
234  cvtColor(*cameraFrame, tmpFrame, CV_BGR2RGB);
235  break;
236  default:
237  cvtColor(*cameraFrame, tmpFrame, CV_BGR2RGB);
238  break;
239 
240  }
241  lastValidFrame = tmpFrame.clone();
242  }
243  else
244  {
245  tmpFrame = lastValidFrame.clone();
246  }
247 
248  // QImage needs the data to be stored continuously in memory
249  assert(tmpFrame.isContinuous());
250 
251  // Assign OpenCV's image buffer to the QImage. Note that the bytesPerLine parameter
252  // (http://qt-project.org/doc/qt-4.8/qimage.html#QImage-6) is 3*width because each pixel
253  // has three bytes.
254  qtCameraFrame = QImage(tmpFrame.data, tmpFrame.cols, tmpFrame.rows, tmpFrame.cols * 3, QImage::Format_RGB888);
255  qtEnhancedFrame = qtCameraFrame;
256 
257  QPainter painter(&qtEnhancedFrame);
258  if (bDisplayCrosshair)
259  {
260  DisplayCrossHairs(qtCameraFrame, painter);
261  }
262 
263 
264  DisplayReceiverValues(qtCameraFrame, painter, detectionData);
265 
266  if (bDisplayTime)
267  {
268  boost::posix_time::ptime myTime(boost::posix_time::microsec_clock::local_time());
269  QString timeStr(boost::posix_time::to_simple_string(myTime).c_str());
270 
271  QRect textRect(0, 0, tmpFrame.cols - 1, tmpFrame.rows - 1);
272  DrawVideoText(qtCameraFrame, painter, textRect, timeStr);
273  }
274 
275  repaint();
276 }

338 bool SortDetectionsInThreatLevel (Detection::Ptr &left, Detection::Ptr &right)
339 
340 {
341  // Sort by:
342  // - Threat Level.
343  // - Distance from vehicle
344  // - Left/Right
345 
346  // Same threat level, sort from left to right
347  if (left->threatLevel == right->threatLevel)
348  {
349  // Same distance from vehicle, sort from left to right
350  if (abs(left->relativeToVehicleCart.bodyRelative.forward - right->relativeToVehicleCart.bodyRelative.forward) < 0.01)
351  {
352  // Remember vehicle Y is positive going left, So we reverse the < operator.
353  if (left->relativeToVehicleCart.bodyRelative.left > right->relativeToVehicleCart.bodyRelative.left)
354  {
355  return(true);
356  }
357  else
358  {
359  return(false);
360  }
361  }
362  // Not same depth, compare forward
363  else if (left->relativeToVehicleCart.bodyRelative.forward < right->relativeToVehicleCart.bodyRelative.forward)
364  {
365  return(true);
366  }
367  else
368  {
369  return(false);
370  }
371  }
372  else if (left->threatLevel < right->threatLevel)
373  {
374  return(true);
375  }
376  else
377  {
378  return(false);
379  }
380 }

452 void VideoViewer::DisplayCrossHairs(QImage &sourceFrame, QPainter &painter)
453 {
454  // Display Camera squares
455  DrawChannelOutlines(sourceFrame, painter);
456 
457  // Display ticks
458 
459  float tickIncrement = 2.5; // 1 tick every 2.5 degrees.
460 
461  // Draw horizontal ticks
462  float longTick = 25; // Tick length in pixels
463  float mediumTick = 15;
464  float shortTick = 5;
465 
466  int tickQty = 1+ ((RAD2DEG(videoCapture->calibration.fovHeight) /2) / tickIncrement); // Add 1 for the tick at position 0
467  for (int tickIndex = -tickQty; tickIndex <= tickQty; tickIndex++)
468  {
469  // Short tick every 2.5 degree.
470  // Medium tick every 5 degree
471  // Long tick every 10 degree
472  float tickLength = shortTick;
473  int tickWidth = 1;
474  if ((tickIndex % 4) == 0)
475  {
476  tickLength = longTick;
477  tickWidth = 2;
478  }
479  else if ((tickIndex % 2) == 0)
480  {
481  tickLength = mediumTick;
482  }
483 
484  DrawHorizontalTicks(sourceFrame, painter, tickIndex * tickIncrement, tickLength, 1 + floor(tickWidth * displayScaleFactor));
485 
486  //CvPoint textPoint(0 /* videoCapture->calibration.frameWidthInPixels - 1*/, videoCapture->calibration.frameHeightInPixels - 1);
487  }
488 
489 
490  // Draw vertical ticks
491  longTick = 25; // Tick length in pixels (not the same as horizontal ticks)
492  mediumTick = 15;
493  shortTick = 8;
494  tickQty = 1+ ((RAD2DEG(videoCapture->calibration.fovWidth) /2) / tickIncrement); // Add 1 for the tick at position 0
495  for (int tickIndex = -tickQty; tickIndex <= tickQty; tickIndex++)
496  {
497  // Short tick every 2.5 degree.
498  // Medium tick every 5 degree
499  // Long tick every 10 degree
500  float tickLength = shortTick;
501  int tickWidth = 1;
502  if ((tickIndex % 4) == 0)
503  {
504  tickLength = longTick;
505  tickWidth = 2;
506  }
507  else if ((tickIndex % 2) == 0)
508  {
509  tickLength = mediumTick;
510  }
511 
512  DrawVerticalTicks(sourceFrame, painter, tickIndex * tickIncrement, tickLength, 1 + floor(tickWidth * displayScaleFactor));
513  }
514 }

612 void VideoViewer::GetDetectionColors(const Detection::Ptr &detection, QColor &colorEnhance, int &iThickness)
613 
614 {
615  bool bFlash = false;
616 
617  long millisecs = (boost::posix_time::microsec_clock::local_time() - startTime).total_milliseconds();
618  if ((millisecs/(flashPeriodMillisec/2)) & 0x01)
619  {
620  bFlash = true;
621  }
622 
623  AlertCondition::ThreatLevel threatLevel = detection->threatLevel;
624 
625  switch (threatLevel)
626  {
627 
628  case AlertCondition::eThreatOutlineOnly:
629  {
630  colorEnhance = rgbEnhancePurple; // Try any contrasting color!
631  iThickness = 2;
632  }
633  break;
634 
635  case AlertCondition::eThreatNone:
636  {
637  colorEnhance = rgbEnhanceBlue; // Blue
638  iThickness = 5;
639  }
640  break;
641 
642  case AlertCondition::eThreatLow:
643  {
644  colorEnhance = rgbEnhanceGreen; // Green
645  iThickness = 5;
646  }
647  break;
648 
649  case AlertCondition::eThreatWarn:
650  {
651  colorEnhance = rgbEnhanceYellow; // Yellow
652  iThickness = 15;
653  if (bFlash) iThickness = 5;
654  }
655  break;
656 
657  case AlertCondition::eThreatCritical:
658  {
659  colorEnhance = rgbEnhanceRed; // Red
660  iThickness = 15;
661  if (bFlash) iThickness = 5;
662  }
663  break;
664 
665  default:
666  {
667  colorEnhance = Qt::black;
668  }
669 
670  } // case
671 }

673 bool VideoViewer::GetChannelRect(const Detection::Ptr &detection, CvPoint &topLeft, CvPoint &topRight, CvPoint &bottomLeft, CvPoint &bottomRight)
674 {
675  AWLSettings *globalSettings = AWLSettings::GetGlobalSettings();
676  bool bSomePointsInFront = false;
677 
678  int receiverID = detection->receiverID;
679  int cameraID = videoCapture->GetCameraID();
680 
681  // Channel description pointer
682  int columns = globalSettings->receiverSettings[receiverID].receiverColumns;
683  int voxelIndex = (detection->cellID.row * columns) + detection->cellID.column;
684  VoxelConfig *voxel= &globalSettings->receiverSettings[receiverID].voxelsConfig[voxelIndex];
685
