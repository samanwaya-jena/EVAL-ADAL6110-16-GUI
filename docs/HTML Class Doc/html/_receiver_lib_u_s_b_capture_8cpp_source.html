96 bool ReceiverLibUSBCapture::OpenCANPort()
97 {
98  int ret = 0;
99 
100  reconnectTime = boost::posix_time::microsec_clock::local_time() + boost::posix_time::milliseconds(reopenPortDelaylMillisec);
101 
102  ret = libusb_init(&context);
103 
104  if (ret)
105  return false;
106 
107  // discover devices
108  libusb_device **list;
109  libusb_device_descriptor descriptor;
110  ssize_t cnt = libusb_get_device_list(NULL, &list);
111  ssize_t i = 0;
112  int err = 0;
113  if (cnt < 0)
114  perror("libusb_get_device_list");
115 
116  int matches = 0;
117 
118  for (i = 0; i < cnt; i++) {
119  libusb_device *device = list[i];
120  err = libusb_get_device_descriptor(device, &descriptor);
121  if (!err && descriptor.idVendor == usbVendorId && descriptor.idProduct == usbProductId) {
122  matches ++;
123  }
124  }
125 
126  //if (matches != (receiverID + 1)) return false;
127  if (matches < (receiverID + 1)) return false;
128 
129  for (i = 0; i < cnt; i++) {
130  libusb_device *device = list[i];
131  err = libusb_get_device_descriptor(device, &descriptor);
132  if (!err && descriptor.idVendor == usbVendorId && descriptor.idProduct == usbProductId) {
133  libusb_device_handle* h;
134  err = libusb_open(device, &h);
135  if (err) {
136  perror("libusb_open_no_access");
137  }
138  else {
139  handle = h;
140  //Claim Interface 0 from the device
141  ret = libusb_claim_interface((libusb_device_handle*)handle, 0);
142 
143  if (!ret) {
144  fprintf(stderr, "%ld %d usb_claim_interface %ld %p succeeded\n", (long) receiverID, matches, (long)i, handle);
145  }
146  else {
147  fprintf(stderr, "%ld %d usb_claim_interface %ld %p error %d\n", (long)receiverID, matches, (long) i, handle, ret);
148  libusb_close((libusb_device_handle*)handle);
149  handle = NULL;
150  }
151  }
152  }
153  if (handle) break;
154  }
155 
156 //printf("out %p\n", handle);
157  if (handle)
158  {
159  int transferred = 0;
160  int received = 0;
161 
162  boost::mutex::scoped_lock rawLock(m_Mutex);
163 
164  // DGG: Clear any outstanding data left in the USB buffers
165  do
166  {
167  char tmp[256];
168  ret = libusb_bulk_transfer((libusb_device_handle*)handle, usbEndPointIn, (unsigned char*)tmp, sizeof(tmp), &received, 100);
169  } while (ret == 0 && received > 0);
170 
171  ReceiverCANMessage msg;
172  msg.id = RECEIVERCANMSG_ID_LIDARQUERY;
173 
174  ReceiverCANMessage resp;
175 
176  transferred = WriteBytes((uint8_t*)& msg, sizeof(msg));
177  if (transferred != sizeof(ReceiverCANMessage))
178  return false;
179 
180  received = ReadBytes((uint8_t*)& resp, sizeof(resp));
181  if (received != sizeof(ReceiverCANMessage))
182  return false;
183 
184  rawLock.unlock();
185 
186  SendSoftwareReset();
187 
188  m_calculatedFrameRate = 0;
189 
190  return true;
191  }
192  else { // if (!handle)
193  return false;
194  }
195 }

248

258
