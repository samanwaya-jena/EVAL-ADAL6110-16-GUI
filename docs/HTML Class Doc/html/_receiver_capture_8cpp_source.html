392 void ReceiverCapture::ProcessCompletedFrame()
393 
394 {
395  boost::mutex::scoped_lock rawLock(GetMutex());
396 
397  // timestamp the currentFrame
398  Timestamp elapsed = GetElapsed();
399 
400  ++m_nbrCompletedFrame;
401 
402  currentFrame->timeStamp = elapsed;
403 
404  // TimeStamp all detections and tracks
405  TimestampTracks(currentFrame);
406 
407  TimestampDetections(currentFrame);
408 
409  // Log Tracks?
410  if (receiverStatus.messageMask.bitFieldData.obstacle)
411  {
412  LogTracks(currentFrame);
413  }
414 
415  // Log Distances?
416  if (receiverStatus.messageMask.bitFieldData.distance_1_4 ||
417  receiverStatus.messageMask.bitFieldData.distance_5_8 ||
418  receiverStatus.messageMask.bitFieldData.intensity_1_4 ||
419  receiverStatus.messageMask.bitFieldData.intensity_5_8 ||
420  receiverStatus.messageMask.bitFieldData.distance_intensity)
421  {
422  LogDistances(currentFrame);
423  }
424 
425  FrameID completedFrameID = currentFrame->GetFrameID();
426  bool bFrameToBePublished = false;
427  if (!bFrameInvalidated)
428  {
429  // Push the current frame in the frame buffer
430  acquisitionSequence->sensorFrames.push_front(currentFrame);
431 
432  // Make sure we do not keep too many of those frames around.
433  // Remove the older frame if we exceed the buffer capacity
434  if (acquisitionSequence->sensorFrames.size() > maximumSensorFrames)
435  {
436  acquisitionSequence->sensorFrames.pop_back();
437  }
438 
439  bFrameToBePublished = true;
440  }
441 
442  // Create a new current frame.
443  FrameID localFrameID = acquisitionSequence->AllocateFrameID();
444  currentFrame = SensorFrame::Ptr(new SensorFrame(receiverID, localFrameID, receiverVoxelQty));
445  bFrameInvalidated = false;
446 
447  rawLock.unlock();
448 
449  if (bFrameToBePublished)
450  {
451  PutNews(completedFrameID);
452  DebugFilePrintf(debugFile, "FrameID- %lu", completedFrameID);
453  }
454  else
455  {
456  DebugFilePrintf(debugFile, "FrameIDUnpublished- %lu", completedFrameID);
457  }
458 
459 }
