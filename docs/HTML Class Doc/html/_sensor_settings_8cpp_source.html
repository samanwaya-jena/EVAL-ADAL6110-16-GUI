90 bool SensorSettings::ReadSettings()
91 {
92  // Create an empty property tree object
93  using boost::property_tree::ptree;
94  propTree.clear();
95 
96  // Load the XML file into the property tree. If reading fails
97  // (cannot open file, parse error), an exception is thrown.
98  read_xml(sSettingsFileName, propTree);
99 
100  // Debug and log file control
101  bWriteDebugFile = propTree.get<bool>("config.debug.enableDebugFile", false);
102  bWriteLogFile = propTree.get<bool>("config.debug.enableLogFile", false);
103  sDebugFileName = propTree.get<std::string>("config.debug.debugFileName", "debug.dbg");
104  sLogFileName = propTree.get<std::string>("config.debug.logFileName", "distanceLog.csv");
105  sDebugAndLogFilePath = propTree.get<std::string>("config.debug.defaultPath", "./");
106 
107  // Receivers
108 
109  int receiverQty = propTree.get<int>("config.receivers.receiverQty", 0);
110  receiverSettings.resize(receiverQty);
111 
112  for (int receiverIndex = 0; receiverIndex < receiverQty; receiverIndex++)
113  {
114  std::string receiverKey = std::string("config.receivers.receiver") + std::to_string(receiverIndex);
115 
116  boost::property_tree::ptree& receiverNode = propTree.get_child(receiverKey);
117 
118  ReceiverSettings* receiverPtr = &receiverSettings[receiverIndex];
119  receiverPtr->sReceiverType = receiverNode.get<std::string>("receiverType");
120  receiverPtr->sReceiverRegisterSet = receiverNode.get<std::string>("receiverRegisterSet");
121  receiverPtr->sReceiverVoxelGeometry = receiverNode.get<std::string>("receiverVoxelGeometry");
122 
123  // Display
124  receiverPtr->displayedRangeMin = receiverNode.get<float>("displayedRangeMin", (float)0.0);
125  receiverPtr->displayedRangeMax = receiverNode.get<float>("displayedRangeMax", (float)60.0);
126 
127  // Get the Channel configuration Node
128  std::string voxelGeometryKey = "config." + receiverPtr->sReceiverVoxelGeometry;
129  boost::property_tree::ptree& voxelGeometryNode = propTree.get_child(voxelGeometryKey);
130  GetVoxelGeometry(voxelGeometryNode, receiverPtr);
131  } // for (int receiverIndex = 0;
132 
133 
134  int cameraQty = propTree.get<int>("config.cameras.cameraQty", 0);
135  cameraSettings.resize(cameraQty);
136  for (int cameraIndex = 0; cameraIndex < cameraQty; cameraIndex++)
137  {
138  std::string cameraKey = std::string("config.cameras.camera") + std::to_string(cameraIndex);
139 
140  boost::property_tree::ptree& cameraNode = propTree.get_child(cameraKey);
141  CameraSettings* cameraPtr = &cameraSettings[cameraIndex];
142 
143  cameraPtr->sCameraName = cameraNode.get<std::string>("cameraName", "0");
144  cameraPtr->sCameraAPI = cameraNode.get<std::string>("cameraAPI", "OpenCV");
145  cameraPtr->cameraFlip = cameraNode.get<bool>("cameraFlip", false);
146  Get2DPoint(cameraNode.get_child("fov"), cameraPtr->cameraFovWidthDegrees, cameraPtr->cameraFovHeightDegrees);
147  cameraPtr->barrelK1 = 0.0;
148  cameraPtr->barrelK2 = 0.0;
149 
150  cameraPtr->barrelK1 = cameraNode.get<float>("barrelCorrectionK1", 0.0);
151  cameraPtr->barrelK2 = cameraNode.get<float>("barrelCorrectionK2", 0.0);
152  }
153 
154  brakingDeceleration = propTree.get<float>("config.dynamicTesting.brakingDeceleration", (float)5.096);
155  travelSpeed = propTree.get<float>("config.dynamicTesting.travelSpeed", (float)33.32);
156 
157  // Alert conditions
158  int alertConditionQty = propTree.get<int>("config.dynamicTesting.alertQty", 0);
159  for (int alertConditionIndex = 0; alertConditionIndex < alertConditionQty; alertConditionIndex++)
160  {
161  std::string alertKey = std::string("config.dynamicTesting.alert") + std::to_string(alertConditionIndex);
162 
163  boost::property_tree::ptree& alertNode = propTree.get_child(alertKey);
164  AlertCondition::Ptr newAlert = AlertCondition::Ptr(new AlertCondition());
165  GetAlertConditions(alertNode, *newAlert);
166  AlertCondition::Store(newAlert);
167  }
168 
169  return(true);
170 }

283 void SensorSettings::GetAlertConditions(boost::property_tree::ptree& alertNode, AlertCondition& alert)
284 
285 {
286  std::string sAlertType = alertNode.get<std::string>("alertType", "Invalid");
287  if (sAlertType.compare("distanceWithin") == 0)
288  {
289  alert.alertType = AlertCondition::eAlertDistanceWithin;
290  }
291  else if (sAlertType.compare("distanceOutside") == 0)
292  {
293  alert.alertType = AlertCondition::eAlertDistanceOutside;
294  }
295  else if (sAlertType.compare("speed") == 0)
296  {
297  alert.alertType = AlertCondition::eAlertSpeed;
298  }
299  else if (sAlertType.compare("acceleration") == 0)
300  {
301  alert.alertType = AlertCondition::eAlertAcceleration;
302  }
303  else if (sAlertType.compare("decelerationToStop") == 0)
304  {
305  alert.alertType = AlertCondition::eAlertDecelerationToStop;
306  }
307  else if (sAlertType.compare("TTC") == 0)
308  {
309  alert.alertType = AlertCondition::eAlertTTC;
310  }
311  else
312  {
313  alert.alertType = AlertCondition::eAlertInvalid;
314  }
315 
316  alert.receiverID = alertNode.get<int>("alertReceiver", 0);
317  alert.alertVoxelMask.wordData = alertNode.get<uint16_t>("alertVoxel", 255);
318  alert.minRange = alertNode.get<float>("alertMin", -std::numeric_limits<float>::max());
319  alert.maxRange = alertNode.get<float>("alertMax", std::numeric_limits<float>::max());
320  alert.threatLevel = (AlertCondition::ThreatLevel) (alertNode.get<int>("alertLevel", AlertCondition::eThreatNone));
321 }

358 void SensorSettings::GetVoxelGeometryArray(boost::property_tree::ptree& voxelGeometryNode, ReceiverSettings* receiverPtr)
359 
360 {
361  float columnsFloat;
362  float rowsFloat;
363  int columns;
364  int rows;
365  float fovX;
366  float fovY;
367  float spacingX;
368  float spacingY;
369  float offsetX;
370  float offsetY;
371 
372  float maxRange, maxAscanRange;
373  uint8_t displayColorRed;
374  uint8_t displayColorGreen;
375  uint8_t displayColorBlue;
376 
377 
378  // Range Wraparound trick
379  Get2DPoint(voxelGeometryNode.get_child("arraySize"), columnsFloat, rowsFloat);
380  columns = (int)columnsFloat;
381  rows = (int)rowsFloat;
382 
383  Get2DPoint(voxelGeometryNode.get_child("arrayFOV"), fovX, fovY);
384  Get2DPoint(voxelGeometryNode.get_child("pixelSpacing"), spacingX, spacingY);
385  Get2DPoint(voxelGeometryNode.get_child("arrayOffset"), offsetX, offsetY);
386 
387  maxRange = voxelGeometryNode.get<float>("maxRange", 60.0);
388  maxAscanRange = voxelGeometryNode.get<float>("maxAscanRange", 60.0);
389 
390  GetColor(voxelGeometryNode.get_child("displayColor"), displayColorRed, displayColorGreen, displayColorBlue);
391 
392  float pixelWidth = (fovX - ((columns - 1) * spacingX)) / columns;
393  float pixelHeight = (fovY - ((rows - 1) * spacingY)) / rows;
394 
395  int voxelQty = columns * rows;
396  receiverPtr->voxelsConfig.resize(voxelQty);
397  receiverPtr->receiverColumns = columns;
398  receiverPtr->receiverRows = rows;
399 
400 
401  for (int voxelIndex = 0; voxelIndex < voxelQty; voxelIndex++)
402  {
403  int row = voxelIndex / columns;
404  std::string sColorKey = std::string("displayColorLine") + std::to_string(row);
405  GetColor(voxelGeometryNode.get_child(sColorKey), displayColorRed, displayColorGreen, displayColorBlue);
406 
407  VoxelConfig* voxelConfigPtr = &receiverPtr->voxelsConfig[voxelIndex];
408  voxelConfigPtr->voxelIndex = voxelIndex;
409 
410  voxelConfigPtr->fovWidth = pixelWidth;
411  voxelConfigPtr->fovHeight = pixelHeight;
412 
413  voxelConfigPtr->maxRange = maxRange;
414  voxelConfigPtr->maxAscanRange = maxAscanRange;
415  voxelConfigPtr->displayColorRed = displayColorRed;
416  voxelConfigPtr->displayColorGreen = displayColorGreen;
417  voxelConfigPtr->displayColorBlue = displayColorBlue;
418  } // for (int voxelIndex = ;
419 }
