188 void ReceiverCANCapture::ParseMessage(ReceiverCANMessage &inMsg)
189 
190 {
191  unsigned long msgID = inMsg.id;
192 #ifdef FORCE_FRAME_RESYNC_PATCH
193  ForceFrameResync(inMsg);
194 #endif //FORCE_FRAME_RESYNC_PATCH
195 
196  //printf ("ParseMessage %02x \n", msgID);
197  if (msgID == RECEIVERCANMSG_ID_SENSORSTATUS)
198  {
199  ParseSensorStatus(inMsg);
200  }
201  else if (msgID == RECEIVERCANMSG_ID_SENSORBOOT)
202  {
203  ParseSensorBoot(inMsg);
204  }
205  else if (msgID == RECEIVERCANMSG_ID_COMPLETEDFRAME)
206  {
207  ProcessCompletedFrame();
208  }
209  else if (msgID == RECEIVERCANMSG_ID_OBSTACLETRACK)
210  {
211  ParseObstacleTrack(inMsg);
212  }
213  else if (msgID == RECEIVERCANMSG_ID_OBSTACLEVELOCITY)
214  {
215  ParseObstacleVelocity(inMsg);
216  }
217  else if (msgID == RECEIVERCANMSG_ID_OBSTACLESIZE)
218  {
219  ParseObstacleSize(inMsg);
220  }
221  else if (msgID == RECEIVERCANMSG_ID_OBSTACLEANGULARPOSITION)
222  {
223  ParseObstacleAngularPosition(inMsg);
224  }
225  else if (msgID >= RECEIVERCANMSG_ID_CHANNELDISTANCE1_FIRST && msgID <= RECEIVERCANMSG_ID_CHANNELDISTANCE1_LAST)
226  {
227  ParseChannelDistance(inMsg);
228  }
229  else if (msgID >= RECEIVERCANMSG_ID_CHANNELDISTANCE2_FIRST && msgID <= RECEIVERCANMSG_ID_CHANNELDISTANCE2_LAST)
230  {
231  ParseChannelDistance(inMsg);
232  }
233  else if (msgID >= RECEIVERCANMSG_ID_CHANNELINTENSITY1_FIRST && msgID <= RECEIVERCANMSG_ID_CHANNELINTENSITY1_LAST)
234  {
235  ParseChannelIntensity(inMsg);
236  }
237  else if (msgID >= RECEIVERCANMSG_ID_CHANNELINTENSITY2_FIRST && msgID <= RECEIVERCANMSG_ID_CHANNELINTENSITY2_LAST)
238  {
239  ParseChannelIntensity(inMsg);
240  }
241  else if (msgID == RECEIVERCANMSG_ID_CHANNELDISTANCEANDINTENSITY)
242  {
243  ParseChannelDistanceAndIntensity(inMsg);
244  }
245  else if (msgID == RECEIVERCANMSG_ID_COMMANDMESSAGE) /* Command */
246  {
247  ParseControlMessage(inMsg);
248  }
249  else
250  {
251  DebugFilePrintf(debugFile, "UnknownMessage %d", msgID);
252  InvalidateFrame();
253  }

497 void ReceiverCANCapture::ParseObstacleTrack(ReceiverCANMessage &inMsg)
498 
499 {
500  boost::mutex::scoped_lock rawLock(GetMutex());
501 
502  uint16_t trackID = *(uint16_t *)&inMsg.data[0];
503 
504  Track::Ptr track = acquisitionSequence->MakeUniqueTrack(currentFrame, trackID);
505 
506  track->firstTimeStamp = currentFrame->timeStamp;
507  track->timeStamp = currentFrame->timeStamp;
508 
509  track->trackChannels.wordData = *(uint8_t *)&inMsg.data[2];
510  track->trackMainVoxel = *(uint16_t *)&inMsg.data[3];
511 
512 
513  // Compatibility patch: AWL-7 sends byte data only, but only one voxel per voxelMask.
514  // Other versions send trackMainVoxel. For AWL-7 track main voxel is 0.
515  // Rebuild trackMainVoxel from AWL 7 data.
516  if (this->receiverVoxelQty == 7)
517  {
518  track->trackMainVoxel = 0;
519  uint16_t voxelMask = 0x01;
520  for (uint16_t voxelIndex = 0; voxelIndex < 8; voxelIndex++)
521  {
522  if (track->trackChannels.wordData & voxelMask)
523  {
524  track->trackMainVoxel = voxelIndex;
525  break;
526  }
527  voxelMask <<= 1;
528  }
529  }
530 
531  // Decode rest of message
532  track->probability = *(uint8_t *)&inMsg.data[5];
533 
534  uint16_t intensity = (*(uint16_t *)&inMsg.data[6]);
535  track->intensity = ConvertIntensityToSNR(intensity);
536 
537  track->part1Entered = true;
538 
539  rawLock.unlock();
540  // Debug and Log messages
541  DebugFilePrintf(debugFile, "Msg %lu - Track %u Val %x %d %f %f", inMsg.id, track->trackID, track->trackChannels, track->trackMainVoxel, track->probability, track->intensity);
542 }
