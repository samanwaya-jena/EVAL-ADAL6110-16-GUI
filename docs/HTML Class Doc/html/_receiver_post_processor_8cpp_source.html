65 bool ReceiverPostProcessor::CompleteTrackInfo(SensorFrame::Ptr currentFrame)
66 {
67  bool bAllTracksComplete = true;
68 
69  // Update the coaslesced tracks
70  Track::Vector::iterator trackIterator = currentFrame->tracks.begin();
71 
72  while (trackIterator != currentFrame->tracks.end())
73  {
74  Track::Ptr track = *trackIterator;
75  if (track->IsComplete())
76  {
77  if (track->velocity > 0)
78  {
79  track->timeToCollision = PredictTimeToCollisionConstant(track->distance, track->velocity);
80  }
81 
82  // Deceleration to stop should eventually include track->acceleration, but right now,
83  // that information is judged unreliable.
84  // Will need adjustement later
85  track->decelerationToStop = - CalculateAccelerationToStop(track->distance, track->velocity, 0);
86 
87  track->threatLevel = AlertCondition::FindTrackThreat(currentFrame->GetReceiverID(), track);
88  } // if (track...
89  else
90  {
91  bAllTracksComplete = false;
92  }
93 
94 
95  trackIterator++;
96  } // while (trackIterator...
97 
98  return (bAllTracksComplete);
99 }

102 bool ReceiverPostProcessor::BuildEnhancedDetectionsFromTracks(ReceiverCapture::Ptr receiver, SensorFrame::Ptr currentFrame, Detection::Vector &outDetections)
103 {
104  SensorSettings *settings = SensorSettings::GetGlobalSettings();
105  ReceiverSettings &receiverSettings = settings->receiverSettings[currentFrame->GetReceiverID()];
106  bool bAllTracksComplete = true;
107 
108  // Make sure that we start from scratch. All output detection vector is cleared.
109  outDetections.clear();
110  std::vector<int> detectionIndex(currentFrame->voxelQty);
111 
112 
113  // In voxel order, recreate individual detections from the tracks.
114  for (int voxelIndex = 0; voxelIndex < currentFrame->voxelQty; voxelIndex++)
115  {
116  VoxelMask voxelMask;
117  voxelMask.wordData = 0x01 << (voxelIndex % 8);
118 
119 
120  // Re-Create detections from the coalesced tracks
121 
122  Track::Vector::iterator trackIterator = currentFrame->tracks.begin();
123  while (trackIterator != currentFrame->tracks.end())
124  {
125  Track::Ptr track = *trackIterator;
126 
127 
128  int trackVoxel = track->trackMainVoxel;
129  CellID cellID(trackVoxel % receiver->receiverColumnQty, trackVoxel / receiver->receiverColumnQty);
130  float trackDistance = track->distance;
131 
132 #if 1 // Process voxel wraparound if set
133  if (receiver->lineWrapAround > 0.0)
134  {
135  int rowIndex = 0;
136  int columnIndex = 0;
137 
138  rowIndex = (int) (track->distance / receiver->lineWrapAround);
139  if (rowIndex > (receiver->receiverRowQty - 1))
140  {
141  rowIndex = 0;
142  }
143 
144  columnIndex = voxelIndex % receiver->receiverColumnQty;
145 
146  trackVoxel = (rowIndex * receiver->receiverColumnQty) + columnIndex;
147  cellID = CellID(trackVoxel % receiver->receiverColumnQty, trackVoxel / receiver->receiverColumnQty);
148 
149  trackDistance = track->distance - (rowIndex * receiver->lineWrapAround);
150  }
151 #endif
152 
153  if ( track->IsComplete() && (trackVoxel == voxelIndex) &&
154  (trackDistance >= receiverSettings.displayedRangeMin) &&
155  (trackDistance <= receiverSettings.voxelsConfig[voxelIndex].maxRange))
156  {
157  Detection::Ptr detection = Detection::Ptr(new Detection(currentFrame->receiverID, cellID, detectionIndex[trackVoxel]++));
158  outDetections.push_back(detection);
159 
160  detection->cellID = cellID;
161  detection->distance = trackDistance;
162 
163  detection->intensity = track->intensity;
164  detection->velocity = track->velocity;
165  detection->acceleration = track->acceleration;
166  detection->probability = track->probability;
167 
168  detection->timeStamp = track->timeStamp;
169  detection->firstTimeStamp = track->timeStamp; // TBD
170  detection->trackID = track->trackID;
171  detection->timeToCollision = track->timeToCollision;
172  detection->decelerationToStop = track->decelerationToStop;
173  detection->threatLevel = track->threatLevel;
174 
175  // Place the coordinates relative to all their respective reference systems
176  TransformationNode::Ptr voxelCoords = SensorCoordinates::GetVoxel(detection->receiverID, trackVoxel);
177  SphericalCoord sphericalPointInChannel(detection->distance, (float) M_SENSORCORE_2, 0);
178  detection->relativeToSensorCart = voxelCoords->ToReferenceCoord(eSensorToReceiverCoord, sphericalPointInChannel);
179  detection->relativeToVehicleCart = voxelCoords->ToReferenceCoord(eSensorToVehicleCoord, sphericalPointInChannel);
180  detection->relativeToWorldCart = voxelCoords->ToReferenceCoord(eSensorToWorldCoord, sphericalPointInChannel);
181 
182  detection->relativeToSensorSpherical = detection->relativeToSensorCart;
183  detection->relativeToVehicleSpherical = detection->relativeToVehicleCart;
184  detection->relativeToWorldSpherical = detection->relativeToWorldCart;
185  } // if (track...
186  else if (!track->IsComplete())
187  {
188  bAllTracksComplete = false;
189  }
190 
191  trackIterator++;
192  } // while (trackIterator...
193  } // for (voxelIndex)
194 
195  return (bAllTracksComplete);
196 }

199 bool ReceiverPostProcessor::GetEnhancedDetectionsFromFrame(ReceiverCapture::Ptr receiver, FrameID inFrameID, Publisher::SubscriberID inSubscriberID, Detection::Vector &detectionBuffer)
200 {
201 
202  SensorFrame::Ptr currentFrame = SensorFrame::Ptr(new SensorFrame(receiver->receiverID, inFrameID, receiver->receiverVoxelQty));
203 
204  // Get a local copy of the currentFrame to proceess;
205  if (!receiver->CopyReceiverFrame(inFrameID, currentFrame, inSubscriberID))
206  {
207  return(false);
208  }
209 
210 
211  // Complete the track information that is not yet processed at the receiver level.
212  if (!CompleteTrackInfo(currentFrame))
213  {
214  DebugFilePrintf("Incomplete frame in UpdateTrackInfo- %lu", inFrameID);
215  }
216 
217  // Build distances from the tracks that were accumulated during the frame
218  if (!BuildEnhancedDetectionsFromTracks(receiver, currentFrame, detectionBuffer))
219  {
220  DebugFilePrintf("Incomplete frame- %lu", inFrameID);
221  }
222 
223  if (detectionBuffer.size() <= 0)
224  return(false);
225  else
226  return(true);
227 }

257 float ReceiverPostProcessor::PredictDistance(float currentDistance, float relativeSpeed, float acceleration, float time)
258 
259 {
260  if (isNAN(acceleration)) acceleration = 0;
261 
262  // Distance of vehicle at "time"
263  float at2 = (acceleration * time * time);
264  float predictedDistance = currentDistance + (relativeSpeed * time) + (0.5f * at2);
265 
266  return (predictedDistance);
267 
268 }

279 float ReceiverPostProcessor::CalculateAccelerationToStop(float currentDistance, float relativeSpeed, float currentAcceleration)
280 
281 {
282  if (isNAN(currentAcceleration)) currentAcceleration = 0;
283 
284  float acceleration = (relativeSpeed * relativeSpeed) / currentDistance;
285  if (relativeSpeed < 0)
286  {
287  acceleration *= -1.0;
288  }
289 
290  acceleration -= currentAcceleration;
291 
292  return(acceleration);
293 }
