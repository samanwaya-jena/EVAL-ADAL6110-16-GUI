246 QSize TableView::unconstrainedTableSize() const
247 
248 {
249  AWLSettings *globalSettings = AWLSettings::GetGlobalSettings();
250 
251  // Calculate the dimensions of the table widget
252  QTableWidget *tableWidget = ui.distanceTable;
253 
254  //Calculate height
255  int rowCount = 0;
256  int tableHeight = 0;
257  int receiverCount = globalSettings->receiverSettings.size();
258  for (int receiverID = 0; receiverID < receiverCount; receiverID++)
259  {
260  rowCount += globalSettings->receiverSettings[receiverID].voxelsConfig.size()*displayedDetectionsPerChannel;
261  } // for receiverID
262  tableHeight = rowCount * (tableWidget->rowHeight(1));
263  tableHeight += tableWidget->horizontalHeader()->height() + 10;
264 
265 
266  // Calculate width
267  int tableWidth =0;
268  for (int column = 0; column < tableWidget->columnCount(); column++)
269  {
270 #if 0
271  tableWidth += columnSettings[column].columnWidth;
272 #else
273  QTableWidgetItem * headerItem = tableWidget->horizontalHeaderItem(column);
274  QFontMetrics fm(headerItem->font());
275  int headerTextWidth = fm.width(headerItem->text());
276  int dataTextWidth = fm.width(columnSettings[column].columnFormat);
277  int headerWidth = std::max(headerTextWidth, dataTextWidth) + headerSpacing;
278  tableWidth += headerWidth;
279 #endif
280  }
281 
282  tableWidth += scrollBarWidth;
283  tableWidth += tableWidget->frameWidth() * 2;
284 
285  // Calculate the size of the frame around it.
286  return (QSize(tableWidth, tableHeight));
287 }

290 void TableView::PrepareTableViews()
291 
292 {
293  AWLSettings *globalSettings = AWLSettings::GetGlobalSettings();
294 
295  QTableWidget *tableWidget = ui.distanceTable;
296  tableWidget->clearContents();
297  tableWidget->setRowCount(1);
298 
299  receiverFirstRow.clear();
300 
301  // Adjust the velocity title to display units
302  bool bDisplayVelocityKmh = (globalSettings->velocityUnits == eVelocityUnitsKMH);
303  if (bDisplayVelocityKmh)
304  {
305  tableWidget->horizontalHeaderItem(eRealTimeVelocityColumn)->setText("km/h");
306  }
307  else
308  {
309  tableWidget->horizontalHeaderItem(eRealTimeVelocityColumn)->setText("m/s");
310  }
311 
312  // Adjust column width to the header text
313  int columnQty = tableWidget->columnCount();
314  int tableWidth = 0;
315  for (int column = 0; column < columnQty; column++)
316  {
317 #if 0
318  tableWidget->setColumnWidth(column, columnSettings[column].columnWidth);
319  tableWidth += columnSettings[column].columnWidth;
320 #else
321  QTableWidgetItem * headerItem = tableWidget->horizontalHeaderItem(column);
322  QFontMetrics fm(headerItem->font());
323  int headerTextWidth = fm.width(headerItem->text());
324  int dataTextWidth = fm.width(columnSettings[column].columnFormat);
325  int headerWidth = std::max(headerTextWidth, dataTextWidth) + headerSpacing;
326 
327  tableWidget->setColumnWidth(column, headerWidth);
328  tableWidth += headerWidth;
329 #endif
330  }
331 
332  // Create the table widgets that will hold the data. If required, add additional rows.
333  int row = 0;
334  int receiverCount = globalSettings->receiverSettings.size();
335  for (int receiverID = 0; receiverID < receiverCount; receiverID++)
336  {
337  // Store the index of the first row for each voxel, for future references
338  receiverFirstRow.push_back(row);
339 
340  int voxelCount = globalSettings->receiverSettings[receiverID].voxelsConfig.size();
341  for (int voxelIndex = 0; voxelIndex < voxelCount; voxelIndex++)
342  {
343  for (int detectionID = 0; detectionID < displayedDetectionsPerChannel; detectionID++)
344  {
345  if (row >= tableWidget->rowCount())
346  {
347  tableWidget->insertRow(tableWidget->rowCount());
348  }
349 
350  // Create the table items
351  for (int column = 0; column < tableWidget->columnCount(); column++)
352  {
353  QTableWidgetItem *newItem = new QTableWidgetItem("");
354  newItem->setTextAlignment(columnSettings[column].alignment);
355  tableWidget->setItem(row, column, newItem);
356  } // for column
357 
358  row++;
359  } // for detectionID
360  } // for voxelIndex
361  } // for receiverID
362 }

370 void TableView::DisplayReceiverValues(const Detection::Vector &inData)
371 {
372  AWLSettings *globalSettings = AWLSettings::GetGlobalSettings();
373 
374  QTableWidget *tableWidget = ui.distanceTable;
375  // Fill the table with blanks
376  int receiverCount = globalSettings->receiverSettings.size();
377  int tableRow = 0;
378  for (int receiverID = 0; receiverID < receiverCount; receiverID++)
379  {
380  int voxelCount = globalSettings->receiverSettings[receiverID].voxelsConfig.size();
381  int columns = globalSettings->receiverSettings[receiverID].receiverColumns;
382 
383  for (int voxelIndex = 0; voxelIndex < voxelCount; voxelIndex++)
384  {
385  CellID cellID(voxelIndex % columns, voxelIndex / columns);
386  for (int detectionID = 0; detectionID < displayedDetectionsPerChannel; detectionID++)
387  {
388  AddDistanceToText(tableRow++, tableWidget, receiverID, cellID, detectionID);
389  } // for detection ID;
390  } // for voxelIndex
391  } // for receiverID
392 
393  // Place the receiver data
394  BOOST_FOREACH(const Detection::Ptr & detection, inData)
395  {
396  if (detection->detectionID < displayedDetectionsPerChannel)
397  {
398  CellID cellID = detection->cellID;
399  int columns = globalSettings->receiverSettings[detection->receiverID].receiverColumns;
400  tableRow = detection->detectionID +
401  (cellID.column * displayedDetectionsPerChannel) +
402  (cellID.row * columns * displayedDetectionsPerChannel) +
403  receiverFirstRow.at(detection->receiverID);
404  AddDistanceToText(tableRow, tableWidget, detection);
405  }
406  }
