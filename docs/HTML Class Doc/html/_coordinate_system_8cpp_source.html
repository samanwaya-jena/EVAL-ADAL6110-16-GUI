101 // Convert from cartesian to spherical
102 // Using formaulas indicated in http://mathworld.wolfram.com/SphericalCoordinates.html
103 // But, since we are using physics convention instead of math convention (as Wolfram), swap theta and phi.
104 
105 SphericalCoord CartesianCoord::ToSpherical(const CartesianCoord &inCartesian)
106 {
107  SphericalCoord spherical;
108 
109  spherical.rho = sqrt((inCartesian.cartesian.x*inCartesian.cartesian.x) + (inCartesian.cartesian.y*inCartesian.cartesian.y) + (inCartesian.cartesian.z * inCartesian.cartesian.z));
110 
111  if (spherical.rho != 0.0)
112  {
113  spherical.theta = acos(inCartesian.cartesian.z/spherical.rho);
114  }
115 
116  if (inCartesian.cartesian.x != 0.0)
117  {
118  spherical.phi = atan(inCartesian.cartesian.y / inCartesian.cartesian.x);
119  }
120 
121  return(spherical);
122 }

189 CartesianCoord SphericalCoord::ToCartesian(const SphericalCoord &inSpherical)
190 {
191  CartesianCoord localCartesian;
192  float sinTheta = sin(inSpherical.theta);
193  localCartesian.cartesian.x = inSpherical.rho * cos(inSpherical.phi) * sinTheta;
194  localCartesian.cartesian.y = inSpherical.rho * sin(inSpherical.phi)* sinTheta;
195  localCartesian.cartesian.z = inSpherical.rho * cos(inSpherical.theta);
196  return(localCartesian);
197 }

328 TransformationMatrix::TransformationMatrix(const CartesianCoord &inCartesian, const Orientation &inOrientation)
329 {
330  float cosAlpha = cos(inOrientation.yaw);
331  float sinAlpha= sin(inOrientation.yaw);
332  float cosBeta = cos(inOrientation.pitch);
333  float sinBeta = sin(inOrientation.pitch);
334  float cosGamma = cos(inOrientation.roll);
335  float sinGamma = sin(inOrientation.roll);
336 
337  matrix[0][0] = cosAlpha * cosBeta;
338  matrix[0][1] = (cosAlpha * sinBeta * sinGamma) - (sinAlpha * cosGamma);
339  matrix[0][2] = (cosAlpha * sinBeta *cosGamma) + (sinAlpha * sinGamma);
340  matrix[0][3] = inCartesian.cartesian.x;
341  matrix[1][0] = sinAlpha * cosBeta;
342  matrix[1][1] = (sinAlpha * sinBeta * sinGamma) + (cosAlpha * cosGamma);
343  matrix[1][2] = (sinAlpha * sinBeta * cosGamma) - (cosAlpha * sinGamma);
344  matrix[1][3] = inCartesian.cartesian.y;
345  matrix[2][0] = -sinBeta;
346  matrix[2][1] = cosBeta * sinGamma;
347  matrix[2][2] = cosBeta * cosGamma;
348  matrix[2][3] = inCartesian.cartesian.z;
349  matrix[3][0] = 0.0;
350  matrix[3][1] = 0.0;
351  matrix[3][2] = 0.0;
352  matrix[3][3] = 1.0;
353 }

464 TransformationMatrix TransformationMatrix::Reverse()
465 {
466  TransformationMatrix destinationMatrix;
467 
468  // Ref: http://www.cse.psu.edu/~rcollins/CSE486/lecture12.pdf
469 
470  // Inverse the rotation sub-matrix
471  destinationMatrix.matrix[0][0] = matrix[0][0]?1/matrix[0][0]:0;
472  destinationMatrix.matrix[0][1] = matrix[1][0]?1/matrix[1][0]:0;
473  destinationMatrix.matrix[0][2] = matrix[2][0]?1/matrix[2][0]:0;
474  destinationMatrix.matrix[1][0] = matrix[0][1]?1/matrix[0][1]:0;
475  destinationMatrix.matrix[1][1] = matrix[1][1]?1/matrix[1][1]:0;
476  destinationMatrix.matrix[1][2] = matrix[2][1]?1/matrix[2][1]:0;
477  destinationMatrix.matrix[2][0] = matrix[0][2]?1/matrix[0][2]:0;
478  destinationMatrix.matrix[2][1] = matrix[1][2]?1/matrix[1][2]:0;
479  destinationMatrix.matrix[2][2] = matrix[2][2]?1/matrix[2][2]:0;
480 
481  // Inverse the transformation matrix and apply the row rotation
482  destinationMatrix.matrix[0][3] = -((destinationMatrix.matrix[0][0]*matrix[0][3])+
483  (destinationMatrix.matrix[0][1]*matrix[1][3])+
484  (destinationMatrix.matrix[0][2]*matrix[2][3]));
485 
486  destinationMatrix.matrix[1][3] = -((destinationMatrix.matrix[1][0]*matrix[0][3])+
487  (destinationMatrix.matrix[1][1]*matrix[1][3])+
488  (destinationMatrix.matrix[1][2]*matrix[2][3]));
489 
490  destinationMatrix.matrix[2][3] = -((destinationMatrix.matrix[2][0]*matrix[0][3])+
491  (destinationMatrix.matrix[2][1]*matrix[1][3])+
492  (destinationMatrix.matrix[2][2]*matrix[2][3]));
493  // Carry over the other matrix items
494  destinationMatrix.matrix[3][0] = matrix[3][0];
495  destinationMatrix.matrix[3][1] = matrix[3][1];
496  destinationMatrix.matrix[3][2] = matrix[3][2];
497  destinationMatrix.matrix[3][3] = matrix[3][3];
498 
499  return(destinationMatrix);
500 }

831  position.vect[3] = 1;

916  cameraMatrix.matrix[3][0]= 0;
917  cameraMatrix.matrix[3][1]=0;

919  cameraMatrix.matrix[3][3]=0;
