147 bool SensorCoordinates::SensorToCameraXY(int receiverID, int voxelIndex, int cameraID, const CameraCalibration &camera, const SphericalCoord &sensorCoord, int &cameraX, int &cameraY)
148 {
149  bool bInFront = false;
150 
151  // Voxel description pointer
152  TransformationNode::Ptr voxelCoords = SensorCoordinates::GetVoxel(receiverID, voxelIndex);
153 
154  // Camera FOV description
155  TransformationNode::Ptr cameraCoords = SensorCoordinates::GetCameras()[cameraID];
156  CartesianCoord coordInWorld = voxelCoords->ToReferenceCoord(eSensorToWorldCoord, sensorCoord); // Convert to world
157  CartesianCoord coordInCameraCart = cameraCoords->FromReferenceCoord(eWorldToCameraCoord, coordInWorld); // Convert to camera
158 
159  bInFront = camera.ToFrameXY(coordInCameraCart, cameraX, cameraY);
160 
161  return (bInFront);
162 }

179 bool SensorCoordinates::BuildCoordinatesFromSettings(boost::property_tree::ptree &propTree)
180 {
181  // If the firstNode was already created, cler it so that we can rebuild the whole coordinate tree.
182 
183  if (firstNode.get()) firstNode.reset();
184 
185  firstNode = TransformationNode::Ptr(new TransformationNode(CartesianCoord(0, 0, 0), Orientation(0, 0, 0)));
186 
187  // Build a transformation matrix for each of the pixels in each of the receivers
188  // Loop for the receivers
189  int receiverQty = propTree.get<int>("config.receivers.receiverQty");
190  for (int receiverID = 0; receiverID < receiverQty; receiverID++)
191  {
192  // Get to the receiver in the configuration tree. It will be our anchor in the loop
193 
194  std::string receiverKey = std::string("config.receivers.receiver") + std::to_string(receiverID);
195  boost::property_tree::ptree &receiverPropNode = propTree.get_child(receiverKey);
196 
197 
198  // Get to the receiver geometry in the configuration tree
199  boost::property_tree::ptree &receiverGeometryPropNode = receiverPropNode.get_child("sensorGeometry");
200 
201  // Make the transformation node and add it to the tree
202  TransformationNode::Ptr receiverGeometryNode = GetGeometryFromPropertyNode(receiverGeometryPropNode);
203  firstNode->AddChild(receiverGeometryNode);
204  receivers.push_back(receiverGeometryNode);
205 
206  std::string sVoxelGeometryKey = receiverPropNode.get<std::string>("receiverVoxelGeometry");
207 
208  // Loop for each individual voxel
209  boost::property_tree::ptree &voxelGeometryPropNode = propTree.get_child(std::string("config.")+sVoxelGeometryKey);
210  int voxelQty = voxelGeometryPropNode.get<int>("voxelQty", -1);
211  if (voxelQty != -1)
212  {
213  for (int voxelID = 0; voxelID < voxelQty; voxelID++)
214  {
215  std::string voxelKey = std::string("voxel") + std::to_string(voxelID);
216  boost::property_tree::ptree &voxelPropNode = voxelGeometryPropNode.get_child(voxelKey);
217 
218  // Make the transformation node and add it to the tree
219  TransformationNode::Ptr voxelGeometryNode = GetGeometryFromVoxelPropertyNode(voxelPropNode);
220  receiverGeometryNode->AddChild(voxelGeometryNode);
221  }
222  }
223  else //Voxel qty == -1
224  {
225  float columnsFloat(0.0);
226  float rowsFloat(0.0);
227  int columns(0);
228  int rows(0);
229  float fovX(0);
230  float fovY(0);
231  float spacingX(0.0);
232  float spacingY(0.0);
233  float offsetX(0.0);
234  float offsetY(0.0);
235 
236  SensorSettings::Get2DPoint(voxelGeometryPropNode.get_child("arraySize"), columnsFloat, rowsFloat);
237  columns = (int)columnsFloat;
238  rows = (int)rowsFloat;
239  SensorSettings::Get2DPoint(voxelGeometryPropNode.get_child("arrayFOV"), fovX, fovY);
240  SensorSettings::Get2DPoint(voxelGeometryPropNode.get_child("pixelSpacing"), spacingX, spacingY);
241  SensorSettings::Get2DPoint(voxelGeometryPropNode.get_child("arrayOffset"), offsetX, offsetY);
242 
243  float pixelWidth = (fovX - ((columns - 1)*spacingX)) / columns;
244  float pixelHeight = (fovY - ((rows - 1)*spacingY)) / rows;
245 
246  int matrixVoxelQty = columns * rows;
247  for (int voxelIndex = 0; voxelIndex < matrixVoxelQty; voxelIndex++)
248  {
249  int column = voxelIndex % columns;
250  int row = voxelIndex / columns;
251 
252  float pixelYaw = DEG2RAD(offsetX + ((fovX / 2) - (pixelWidth / 2)) - (column * (pixelWidth + spacingX)));
253  float pixelPitch = DEG2RAD(offsetY - ((fovY / 2) - (pixelHeight / 2)) + (row * (pixelHeight + spacingY)));
254  float pixelRoll = 0.0;
255 
256  Orientation orientation(pixelRoll, pixelPitch, pixelYaw);
257 
258  // Simplification: We assume all voxel sensors are at position 0.0
259  CartesianCoord position(0, 0, 0);
260 
261  // Make the transformation node
262  TransformationNode::Ptr voxelGeometryNode = TransformationNode::Ptr(new TransformationNode(position, orientation));
263  receiverGeometryNode->AddChild(voxelGeometryNode);
264  } // for (int voxelIndex = ;
265  }
266  }
267 
268 
269  // Build a transformation matrix for the cameras
270  // Loop for the cameras
271  int cameraQty = propTree.get<int>("config.cameras.cameraQty");
272  for (int cameraID = 0; cameraID < cameraQty; cameraID++)
273  {
274  std::string cameraKey = std::string("config.cameras.camera") + std::to_string(cameraID);
275 
276  boost::property_tree::ptree &cameraPropNode = propTree.get_child(cameraKey);
277  TransformationNode::Ptr cameraGeometryNode = GetGeometryFromPropertyNode(cameraPropNode);
278 
279  firstNode->AddChild(cameraGeometryNode);
280  cameras.push_back(cameraGeometryNode);
281  }
282 
283 
284  return(true);
285 }

289 TransformationNode::Ptr SensorCoordinates::GetGeometryFromPropertyNode(boost::property_tree::ptree &propNode)
290 
291 {
292  // Read the geometry information and transform it into proper transformation node description
293  // Note that some of the configuration properties are in dregrees and will need to be converted in radians.
294 
295  CartesianCoord position(0, 0, 0);
296  float rollDegree = 0.0; // Read from ini file in degrees
297  float pitchDegree = 0.0; // Read from ini file in degrees
298  float yawDegree = 0.0; // Read from ini file in degrees
299 
300  SensorSettings::GetGeometry(propNode,
301  position.bodyRelative.forward, position.bodyRelative.left, position.bodyRelative.up,
302  pitchDegree, yawDegree, rollDegree);
303 
304  Orientation orientation(DEG2RAD(rollDegree),
305  DEG2RAD(pitchDegree),
306  DEG2RAD(yawDegree));
307 
308  // Make the transformation node
309  TransformationNode::Ptr destNode = TransformationNode::Ptr(new TransformationNode(position, orientation));
310 
311  return (destNode);
312 }

314 TransformationNode::Ptr SensorCoordinates::GetGeometryFromVoxelPropertyNode(boost::property_tree::ptree &voxelNode)
315 
316 {
317  float centerY(0.0);
318  float centerX(0.0);
319  float roll(0.0);
320 
321  // Read the orientation from the configuration file.
322  SensorSettings::GetOrientation(voxelNode.get_child("orientation"), centerY, centerX, roll);
323 
324  roll = 0.0;
325  float pitch = DEG2RAD(centerY);
326  float yaw = DEG2RAD(centerX);
327  Orientation orientation(roll, pitch, yaw);
328 
329  // Simplification: We assume all voxel sensors are at position 0.0
330  CartesianCoord position(0, 0, 0);
331 
332  // Make the transformation node
333  TransformationNode::Ptr destNode = TransformationNode::Ptr(new TransformationNode(position, orientation));
334 
335  return (destNode);
336 }
