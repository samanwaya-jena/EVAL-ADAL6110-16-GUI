80 static float trackDistance = 0.0;
81 static float trackDirection = 1.0; // Set to -1.0 for incoming displacement
82 
83 void ReceiverSimulatorCapture::DoOneThreadIteration()
84 
85 {
86  if (!WasStopped())
87  {
88  // Simulate some tracks for debug purposes
89  Timestamp elapsed = GetElapsed();
90 
91  trackDistance += (0.0001f * trackDirection);
92  if (trackDistance > 10.0) trackDistance = 0.0;
93  if (trackDistance < 0.0) trackDistance = 10.0;
94 
95  for (uint16_t voxelIndex= 0; voxelIndex < 16; voxelIndex++)
96  {
97  for (uint16_t detection = 0; detection < 8; detection++)
98  {
99  // Only display even on even lines and odd on odd lines
100  size_t lineIndex = (voxelIndex / this->receiverColumnQty);
101  size_t lineQty = (GetVoxelQty() / this->receiverColumnQty);
102  if (lineIndex == (voxelIndex%lineQty))
103  {
104  Track::Ptr track = acquisitionSequence->MakeUniqueTrack(currentFrame, (voxelIndex * 8) + detection);
105  track->firstTimeStamp = currentFrame->timeStamp;
106 
107  track->timeStamp = currentFrame->timeStamp;
108  track->distance = (detection * 2) + trackDistance;
109  track->intensity = 1.00;
110  track->trackMainVoxel = voxelIndex;
111 
112  track->trackChannels.wordData = 0x01 << (track->trackMainVoxel % 8);
113 
114 
115  track->velocity = 3.0f * trackDirection;
116  track->acceleration = 0;
117  track->threatLevel = AlertCondition::eThreatLow;
118  track->part1Entered = true;
119  track->part2Entered = true;
120  track->part3Entered = true;
121  track->part4Entered = true;
122 
123  track->probability = 99;
124  track->timeStamp = elapsed;
125  track->firstTimeStamp = elapsed;
126  }
127  }
128  }
129 
130  threadCount = ++ threadCount % 7;
131  ProcessCompletedFrame();
132  } // if (!WasStoppped)
133 }
