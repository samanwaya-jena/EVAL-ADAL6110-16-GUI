135 void VideoCapture::DoThreadLoop()
136 
137 {
138  while (!WasStopped())
139  {
140 
141  // Acquire one frame from camera source or AVI

143  // Do not lock the thread. We should not be blocking
144  // boost::mutex::scoped_lock threadLock(GetMutex());
145 
146  DoThreadIteration();
147

150  // Messages must be at leat 1ms apart.

163 void VideoCapture::DoThreadIteration()
164 
165 {
166 
167  // If camera is not opened, check to see if we must try to reopen
168  if (!cam.isOpened() && boost::posix_time::microsec_clock::local_time() > reconnectTime)
169  {
170  OpenCamera();
171  reconnectTime = boost::posix_time::microsec_clock::local_time() + boost::posix_time::milliseconds(reopenCameraDelaylMillisec);
172  return;
173  }
174 
175  // Acquire from camera source or AVI
176  if( cam.isOpened() )
177  {
178  // Set to all zeros. This will be used to see if we have captured something, since normal openCV validations
179  // don't seem to work.
180  bufferFrame.setTo(0);
181 
182  // cam.read returns true, even when camera is unplugged. But, in this last case, the bufferFrame is
183  // all zeros so we need to test for both
184  if ((!cam.read(bufferFrame)) || (norm(bufferFrame)==0))
185  {
186  cam.release();
187  currentFrame.create(calibration.frameHeightInPixels, calibration.frameWidthInPixels, CV_8UC3);
188  currentFrame.setTo(cv::Scalar(0, 0, 0));;
189  return;
190  }
191 
192  // cvQueryFrame (cam.read) is blocking, while waiting for the frame.
193  //Check again if we were stopped in the meantime
194  if (WasStopped())
195  {
196  return;
197  }
198 
199  if (!bufferFrame.empty())
200  {
201  boost::mutex::scoped_lock currentLock(GetMutex());
202  if (bCameraFlip)
203  {
204  cv::flip(bufferFrame, currentFrame, -1);
205  }
206  else
207  {
208  bufferFrame.copyTo(currentFrame);
209  }
210  currentLock.unlock();
211  PutNews();
212 
213  } // if (!bufferFrame.empty())
214 
215  reconnectTime = boost::posix_time::microsec_clock::local_time()+boost::posix_time::milliseconds(reopenCameraDelaylMillisec);
216  } // if( cam.isOpened() )
217  else
218  {
219  // No cam opened. Show blank image
220 
221  boost::mutex::scoped_lock currentLock(GetMutex());
222 
223  if (currentFrame.empty())
224  {
225  currentFrame.create(calibration.frameHeightInPixels, calibration.frameWidthInPixels, CV_8UC3);
226  //currentFrame.ones(calibration.frameHeightInPixels, calibration.frameWidthInPixels,CV_8UC3);
227  currentFrame.setTo(cv::Scalar(0, 0, 0));
228 
229  }
230  currentLock.unlock();
231  PutNews();
232  }
233 }

256 bool VideoCapture::OpenCamera()
257 
258 {
259  int inputID = 0;
260 
261 
262  if (!sCameraName.empty()) inputID = atoi(sCameraName.c_str());
263 
264  // Determine capture source: Camera, Single Frame or AVI
265  if( sCameraName.empty() || isdigit(sCameraName.c_str()[0]) )
266  {
267  cam.open(inputID);
268  }
269  else
270  {
271  cam.open(sCameraName);
272  }
273 
274 
275  if (cam.isOpened())
276  {
277  calibration.frameWidthInPixels = (int) cam.get(cv::CAP_PROP_FRAME_WIDTH);
278  calibration.frameHeightInPixels = (int) cam.get(cv::CAP_PROP_FRAME_HEIGHT);
279  float framesPerSecond = (float) cam.get(cv::CAP_PROP_FPS);
280 
281  if (framesPerSecond < 1) framesPerSecond = FRAME_RATE; // cv::CAP_PROP_FPS may reurn 0;
282  frameRate = (float) 1.0/framesPerSecond;
283  reconnectTime = boost::posix_time::microsec_clock::local_time()+boost::posix_time::milliseconds(reopenCameraDelaylMillisec);
284 
285  return(true);
286  }
287  else // No camera
288  {
289  // Set default values not to be zeroes
290  calibration.frameWidthInPixels = 640;
291  calibration.frameHeightInPixels = 480;
292 
293  frameRate = 1.0f/30.0f;
294  reconnectTime = boost::posix_time::microsec_clock::local_time()+boost::posix_time::milliseconds(reopenCameraDelaylMillisec);
295 
296  return(false);
297  }
298 }

300 bool VideoCapture::ReadConfigFromPropTree(boost::property_tree::ptree &propTree)
301 {
302  std::string cameraKey = std::string("config.cameras.camera") + std::to_string(cameraID);
303 
304  boost::property_tree::ptree &cameraNode = propTree.get_child(cameraKey);
305 
306  sCameraName = cameraNode.get<std::string>("cameraName");
307  bCameraFlip = cameraNode.get<bool>("cameraFlip");
308 
309 
311  float cameraFovWidthDegrees;
312  float cameraFovHeightDegrees;
313  AWLSettings::Get2DPoint(cameraNode.get_child("fov"), cameraFovWidthDegrees, cameraFovHeightDegrees);
314 
315  calibration.fovWidth = DEG2RAD(cameraFovWidthDegrees);
316  calibration.fovHeight = DEG2RAD(cameraFovHeightDegrees);
317  calibration.CalculateFocalLengthsFromFOVs();
318 
319  calibration.centerX = cameraNode.get<float>("centerCorrectionX", 0.0);
320  calibration.centerY = cameraNode.get<float>("centerCorrectionY", 0.0);
321  calibration.radialK1 = cameraNode.get<float>("radialCorrectionK1", 0.0);
322  calibration.radialK1 = cameraNode.get<float>("radialCorrectionK1", 0.0);
323  calibration.radialK2 = cameraNode.get<float>("radialCorrectionK2", 0.0);
324  calibration.radialK3 = cameraNode.get<float>("radialCorrectionK3", 0.0);
325  calibration.tangentialP1 = cameraNode.get<float>("tangentialCorrectionP1", 0.0);
326  calibration.tangentialP1 = cameraNode.get<float>("tangentialCorrectionP2", 0.0);
327 
328  return(true);
329 }
